118 static bool _isPlyfile( const std::string& filename )
119 {
120  const size_t size = filename.length();
121  if( size < 5 )
122  return false;
123 
124  if( filename[size-4] != '.' || filename[size-3] != 'p' ||
125  filename[size-2] != 'l' || filename[size-1] != 'y' )
126  {
127  return false;
128  }
129  return true;
130 }

133 void Config::_loadModels()
134 {
135  if( !_models.empty( )) // only load on the first config run
136  return;
137 
138  eq::Strings filenames = _initData.getFilenames();
139  while( !filenames.empty( ))
140  {
141  const std::string filename = filenames.back();
142  filenames.pop_back();
143 
144  if( _isPlyfile( filename ))
145  {
146  Model* model = new Model;
147 
148  if( _initData.useInvertedFaces() )
149  model->useInvertedFaces();
150 
151  if( !model->readFromFile( filename.c_str( )))
152  {
153  LBWARN << "Can't load model: " << filename << std::endl;
154  delete model;
155  }
156  else
157  _models.push_back( model );
158  }
159  else
160  {
161  const std::string basename = lunchbox::getFilename( filename );
162  if( basename == "." || basename == ".." )
163  continue;
164 
165  // recursively search directories
166  const eq::Strings subFiles = lunchbox::searchDirectory( filename,
167  ".*" );
168  for(eq::StringsCIter i = subFiles.begin(); i != subFiles.end(); ++i)
169  filenames.push_back( filename + '/' + *i );
170  }
171  }
172 }

174 void Config::_registerModels()
175 {
176  // Register distribution helpers on each config run
177  const bool createDist = _modelDist.empty(); //first run, create distributors
178  const size_t nModels = _models.size();
179  LBASSERT( createDist || _modelDist.size() == nModels );
180 
181  for( size_t i = 0; i < nModels; ++i )
182  {
183  const Model* model = _models[i];
184  ModelDist* modelDist = 0;
185  if( createDist )
186  {
187  modelDist = new ModelDist( model );
188  _modelDist.push_back( modelDist );
189  }
190  else
191  modelDist = _modelDist[i];
192 
193  modelDist->registerTree( getClient( ));
194  LBASSERT( modelDist->isAttached() );
195 
196  _frameData.setModelID( modelDist->getID( ));
197  }
198 
199  LBASSERT( _modelDist.size() == nModels );
200 
201  if( !_modelDist.empty( ))
202  {
203  ModelAssigner assigner( _modelDist );
204  accept( assigner );
205  }
206 }

245 const Model* Config::getModel( const eq::uint128_t& modelID )
246 {
247  if( modelID == 0 )
248  return 0;
249 
250  // Protect if accessed concurrently from multiple pipe threads
251  const eq::Node* node = getNodes().front();
252  const bool needModelLock = (node->getPipes().size() > 1);
253  lunchbox::ScopedWrite _mutex( needModelLock ? &_modelLock : 0 );
254 
255  const size_t nModels = _models.size();
256  LBASSERT( _modelDist.size() == nModels );
257 
258  for( size_t i = 0; i < nModels; ++i )
259  {
260  const ModelDist* dist = _modelDist[ i ];
261  if( dist->getID() == modelID )
262  return _models[ i ];
263  }
264 
265  _modelDist.push_back( new ModelDist );
266  Model* model = _modelDist.back()->loadModel( getApplicationNode(),
267  getClient(), modelID );
268  LBASSERT( model );
269  _models.push_back( model );
270 
271  return model;
272 }

283 void Config::_updateData()
284 {
285  // update camera
286  if( _animation.isValid( ))
287  {
288  const eq::Vector3f& modelRotation = _animation.getModelRotation();
289  const CameraAnimation::Step& curStep = _animation.getNextStep();
290 
291  _frameData.setModelRotation( modelRotation);
292  _frameData.setRotation( curStep.rotation );
293  _frameData.setCameraPosition( curStep.position );
294  }
295  else
296  {
297  if( _frameData.usePilotMode())
298  _frameData.spinCamera( -0.001f * _spinX, -0.001f * _spinY );
299  else
300  _frameData.spinModel( -0.001f * _spinX, -0.001f * _spinY, 0.f );
301 
302  _frameData.moveCamera( 0.0f, 0.0f, 0.001f * _advance );
303  }
304 
305  // idle mode
306  if( isIdleAA( ))
307  {
308  LBASSERT( _numFramesAA > 0 );
309  _frameData.setIdle( true );
310  }
311  else
312  _frameData.setIdle( false );
313 
314  _numFramesAA = 0;
315 }

347 bool Config::handleEvent( const eq::ConfigEvent* event )
348 {
349  switch( event->data.type )
350  {
351  case eq::Event::KEY_PRESS:
352  {
353  if( _handleKeyEvent( event->data.keyPress ))
354  {
355  _redraw = true;
356  return true;
357  }
358  break;
359  }
360 
361  case eq::Event::CHANNEL_POINTER_BUTTON_PRESS:
362  {
363  const eq::uint128_t& viewID = event->data.context.view.identifier;
364  _frameData.setCurrentViewID( viewID );
365  if( viewID == 0 )
366  {
367  _currentCanvas = 0;
368  return false;
369  }
370 
371  const View* view = _getCurrentView();
372  const eq::Layout* layout = view->getLayout();
373  const eq::Canvases& canvases = getCanvases();
374  for( eq::CanvasesCIter i = canvases.begin();
375  i != canvases.end(); ++i )
376  {
377  eq::Canvas* canvas = *i;
378  const eq::Layout* canvasLayout = canvas->getActiveLayout();
379 
380  if( canvasLayout == layout )
381  {
382  _currentCanvas = canvas;
383  return true;
384  }
385  }
386  return true;
387  }
388 
389  case eq::Event::CHANNEL_POINTER_BUTTON_RELEASE:
390  {
391  const eq::PointerEvent& releaseEvent =
392  event->data.pointerButtonRelease;
393  if( releaseEvent.buttons == eq::PTR_BUTTON_NONE)
394  {
395  if( releaseEvent.button == eq::PTR_BUTTON1 )
396  {
397  _spinX = releaseEvent.dy;
398  _spinY = releaseEvent.dx;
399  _redraw = true;
400  return true;
401  }
402  if( releaseEvent.button == eq::PTR_BUTTON2 )
403  {
404  _advance = -releaseEvent.dy;
405  _redraw = true;
406  return true;
407  }
408  }
409  break;
410  }
411  case eq::Event::CHANNEL_POINTER_MOTION:
412  {
413  switch( event->data.pointerMotion.buttons )
414  {
415  case eq::PTR_BUTTON1:
416  _spinX = 0;
417  _spinY = 0;
418 
419  if( _frameData.usePilotMode())
420  _frameData.spinCamera(
421  -0.005f * event->data.pointerMotion.dy,
422  -0.005f * event->data.pointerMotion.dx );
423  else
424  _frameData.spinModel(
425  -0.005f * event->data.pointerMotion.dy,
426  -0.005f * event->data.pointerMotion.dx, 0.f );
427  _redraw = true;
428  return true;
429 
430  case eq::PTR_BUTTON2:
431  _advance = -event->data.pointerMotion.dy;
432  _frameData.moveCamera( 0.f, 0.f, .005f * _advance );
433  _redraw = true;
434  return true;
435 
436  case eq::PTR_BUTTON3:
437  _frameData.moveCamera( .0005f * event->data.pointerMotion.dx,
438  -.0005f * event->data.pointerMotion.dy,
439  0.f );
440  _redraw = true;
441  return true;
442  }
443  break;
444  }
445 
446  case eq::Event::CHANNEL_POINTER_WHEEL:
447  {
448  _frameData.moveCamera( -0.05f * event->data.pointerWheel.yAxis,
449  0.f,
450  0.05f * event->data.pointerWheel.xAxis );
451  _redraw = true;
452  return true;
453  }
454 
455  case eq::Event::MAGELLAN_AXIS:
456  {
457  _spinX = 0;
458  _spinY = 0;
459  _advance = 0;
460  _frameData.spinModel( 0.0001f * event->data.magellan.xRotation,
461  0.0001f * event->data.magellan.yRotation,
462  0.0001f * event->data.magellan.zRotation );
463  _frameData.moveCamera( 0.0001f * event->data.magellan.xAxis,
464  0.0001f * event->data.magellan.yAxis,
465  0.0001f * event->data.magellan.zAxis );
466  _redraw = true;
467  return true;
468  }
469 
470  case eq::Event::MAGELLAN_BUTTON:
471  {
472  if( event->data.magellan.button == eq::PTR_BUTTON1 )
473  _frameData.toggleColorMode();
474 
475  _redraw = true;
476  return true;
477  }
478 
479  case eq::Event::WINDOW_EXPOSE:
480  case eq::Event::WINDOW_RESIZE:
481  case eq::Event::WINDOW_CLOSE:
482  case eq::Event::VIEW_RESIZE:
483  _redraw = true;
484  break;
485 
486  default:
487  break;
488  }
489 
490  _redraw |= eq::Config::handleEvent( event );
491  return _redraw;
492 }

797 co::uint128_t Config::sync( const co::uint128_t& version )
