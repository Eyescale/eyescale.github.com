59 void Channel::frameDraw( const eq::uint128_t& frameID )
60 {
61  // setup OpenGL State
62  eq::Channel::frameDraw( frameID );
63 
64  // - 2D viewport
65  Window *window = static_cast< Window* >( getWindow( ));
66  osg::ref_ptr< SceneView > view = window->getSceneView();
67 
68  const eq::PixelViewport& pvp = getPixelViewport();
69  view->setViewport( pvp.x, pvp.y, pvp.w, pvp.h );
70 
71  // - Stereo
72  view->setDrawBufferValue( getDrawBuffer( ));
73  const eq::ColorMask& colorMask = getDrawBufferMask();
74 
75  osgUtil::RenderStage* stage = view->getRenderStage();
76  osg::ref_ptr< osg::ColorMask > osgMask = stage->getColorMask();
77  osgMask->setMask( colorMask.red, colorMask.green, colorMask.blue, true );
78 
79  // - Frustum (Projection matrix)
80  const eq::Frustumf& frustum = getFrustum();
81  view->setProjectionMatrixAsFrustum(
82  frustum.left(), frustum.right(), frustum.bottom(), frustum.top(),
83  frustum.near_plane(), frustum.far_plane( ));
84 
85  // - Camera (Model Matrix)
86  const Pipe *pipe = static_cast< const Pipe* >( getPipe( ));
87  const FrameData& frameData = pipe->getFrameData();
88 
89  const eq::Vector3f position = frameData.getCameraPosition();
90  const eq::Vector3f lookAt = frameData.getCameraLookAtPoint();
91  const eq::Vector3f upVector = frameData.getCameraUpVector();
92 
93  const osg::Vec3f pos( position.x(), position.y(), position.z( ));
94  const osg::Vec3f look( lookAt.x(), lookAt.y(), lookAt.z( ));
95  const osg::Vec3f up( upVector.x(), upVector.y(), upVector.z( ));
96 
97  view->setViewMatrixAsLookAt( pos, look, up );
98 
99  // - Frustum position (View Matrix)
100  osg::Matrix headView = view->getViewMatrix();
101  headView.postMult( vmmlToOsg( getHeadTransform( )));
102  view->setViewMatrix( headView );
103 
104  // - Render
105  view->cull();
106  view->draw();
107 }
