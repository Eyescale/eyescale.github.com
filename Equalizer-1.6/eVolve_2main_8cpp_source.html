55 int main( const int argc, char** argv )
56 {
57  // 1. Equalizer initialization
58  NodeFactory nodeFactory;
59  eVolve::initErrors();
60 
61  if( !eq::init( argc, argv, &nodeFactory ))
62  {
63  LBERROR << "Equalizer init failed" << std::endl;
64  eVolve::exitErrors();
65  return EXIT_FAILURE;
66  }
67 
68  // 2. parse arguments
69  eVolve::LocalInitData initData;
70  initData.parseArguments( argc, argv );
71 
72  // 3. initialization of local client node
73  lunchbox::RefPtr< eVolve::EVolve > client = new eVolve::EVolve( initData );
74  if( !client->initLocal( argc, argv ))
75  {
76  LBERROR << "Can't init client" << std::endl;
77  eq::exit();
78  eVolve::exitErrors();
79  return EXIT_FAILURE;
80  }
81 
82  // 4. run client
83  const int ret = client->run();
84 
85  // 5. cleanup and exit
86  client->exitLocal();
87 
88  LBASSERTINFO( client->getRefCount() == 1, "Client still referenced by " <<
89  client->getRefCount() - 1 );
90  client = 0;
91 
92  eq::exit();
93  eVolve::exitErrors();
94  return ret;
95 }
