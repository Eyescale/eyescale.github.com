00091     const eq::Vector3f upVector = frameData.getCameraUpVector();

00095     const osg::Vec3f up( upVector.x(), upVector.y(), upVector.z( ));
00096   
00097     view->setViewMatrixAsLookAt( pos, look, up );

00099     // - Frustum position (View Matrix)
