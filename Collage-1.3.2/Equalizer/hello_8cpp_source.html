00082     for( int i = 0; i < 6; i++ )
00083     {
00084         glColor3f( (i&1) ? 0.5f:1.0f, (i&2) ? 1.0f:0.5f, (i&4) ? 1.0f:0.5f );

00086         glNormal3f( 0.0f, 0.0f, 1.0f );

00092         glEnd();

00094         if( i < 3 )
