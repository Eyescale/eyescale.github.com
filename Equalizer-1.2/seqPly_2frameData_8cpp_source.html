00240 void FrameData::reset()
00241 {
00242     eq::Matrix4f model = eq::Matrix4f::IDENTITY;
00243     model.rotate_x( static_cast<float>( -M_PI_2 ));
00244     model.rotate_y( static_cast<float>( -M_PI_2 ));
00245 
00246     if( _position == eq::Vector3f( 0.f, 0.f, -2.f ) && 
00247         _rotation == eq::Matrix4f::IDENTITY && _modelRotation == model )

00253         _position   = eq::Vector3f::ZERO;
