256  template< class V > V _get( const std::string& k ) const
257  {
258  if( !boost::has_trivial_assign< V >( ))
259  LBTHROW( std::runtime_error( "Can't retrieve non-POD " +
260  className( V( ))));
261  if( boost::is_pointer< V >::value )
262  LBTHROW( std::runtime_error( "Can't retrieve pointers" ));
263 
264  const std::string& value = (*this)[ k ];
265  if( value.size() != sizeof( V ))
266  LBTHROW( std::runtime_error( std::string( "Wrong value size " ) +
267  boost::lexical_cast< std::string >( value.size( )) +
268  " for type " + className( V( ))));
269 
270  V v( *reinterpret_cast< const V* >( value.data( )));
271  if( _swap( ))
272  byteswap( v );
