00158 void Window::_loadShaders()
00159 {
00160     if( _state->getShader( vertexShader_glsl ) !=
00161         VertexBufferState::INVALID )
00162     {
00163         // already loaded
00164         return;
00165     }
00166 
00167     // Check if functions are available
00168     if( !GLEW_VERSION_2_0 )
00169     {
00170         LBWARN << "Shader function pointers missing, using fixed function "
00171                << "pipeline" << std::endl;
00172         return;
00173     }
00174 
00175     const GLuint vShader = _state->newShader( vertexShader_glsl,
00176                                               GL_VERTEX_SHADER );
00177     LBASSERT( vShader != VertexBufferState::INVALID );
00178     const GLchar* vShaderPtr = vertexShader_glsl;
00179     glShaderSource( vShader, 1, &vShaderPtr, 0 );
00180     glCompileShader( vShader );
00181 
00182     GLint status;
00183     glGetShaderiv( vShader, GL_COMPILE_STATUS, &status );
00184     if( !status )
00185     {
00186         LBWARN << "Failed to compile vertex shader" << std::endl;
00187         return;
00188     }
00189     
00190     const GLuint fShader = 
00191         _state->newShader( fragmentShader_glsl, GL_FRAGMENT_SHADER );
00192     LBASSERT( fShader != VertexBufferState::INVALID );
00193     const GLchar* fShaderPtr = fragmentShader_glsl;
00194     glShaderSource( fShader, 1, &fShaderPtr, 0 );
00195     glCompileShader( fShader );
00196     glGetShaderiv( fShader, GL_COMPILE_STATUS, &status );
00197     if( !status )
00198     {
00199         LBWARN << "Failed to compile fragment shader" << std::endl;
00200         return;
00201     }
00202     
00203     const GLuint program = _state->newProgram( getPipe() );
00204     LBASSERT( program != VertexBufferState::INVALID );
00205     glAttachShader( program, vShader );
00206     glAttachShader( program, fShader );
00207     glLinkProgram( program );
00208     glGetProgramiv( program, GL_LINK_STATUS, &status );
00209     if( !status )
00210     {
00211         LBWARN << "Failed to link shader program" << std::endl;
00212         return;
00213     }
00214     
00215     // turn off OpenGL lighting if we are using our own shaders
00216     glDisable( GL_LIGHTING );
00217 
00218     LBINFO << "Shaders loaded successfully" << std::endl;
00219 }
