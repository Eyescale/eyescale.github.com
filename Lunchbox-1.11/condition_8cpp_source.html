106 bool Condition::timedWait( const uint32_t timeout )
107 {
108  if( timeout == LB_TIMEOUT_INDEFINITE )
109  {
110  wait();
111  return true;
112  }
113 
114  const uint32_t time = timeout == LB_TIMEOUT_DEFAULT ?
115  300000 /* 5 min */ : timeout;
116 
117 #ifdef _WIN32
118  int error = pthread_cond_timedwait_w32_np( &_impl->cond, &_impl->mutex,
119  time );
120 #else
121  const timespec delta = convertToTimespec( time );
122  timeval now;
123  gettimeofday( &now, 0 );
124 
125  timespec then;
126  then.tv_sec = delta.tv_sec + now.tv_sec;
127  then.tv_nsec = delta.tv_nsec + now.tv_usec * 1000;
128  while( then.tv_nsec > 1000000000 )
129  {
130  ++then.tv_sec;
131  then.tv_nsec -= 1000000000;
132  }
133 
134  int error = pthread_cond_timedwait( &_impl->cond, &_impl->mutex, &then );
135 #endif
136  if( error == ETIMEDOUT )
137  return false;
138 
139  if( error )
140  LBERROR << "pthread_cond_timedwait failed: " << strerror( error )
141  << std::endl;
142  return true;
143 }
