225  vector[ i ] = i;
226  }

228

230  typename V::const_iterator it = vector.begin();
231  for( size_t i = 0; i < LOOPSIZE*10; ++i )
232  {
233  const size_t value = *it;
234  TESTINFO( i == value || value == 0, i << ", " << value );
235  ++it;
236  if( it == vector.end( ))
237  it = vector.begin();
238  }
239  rTime_ = _clock.getTimef();
240 
241  _clock.reset();
242  for( size_t i = 0; i < 10; ++i )
243  {
244  V copy;
245  copy = vector;
246  vector = copy;
247  TEST( copy.size() >= vector.size( ));
248  }
249  cTime_ = _clock.getTimef();
250 
251  _clock.reset();
252  const size_t pos = vector.size() / 2;
253  typename V::iterator i = vector.begin() + pos;
254  typename V::iterator j = vector.erase( i );
255  TEST( j != vector.end( ));
256  TEST( *j == 0 || *j >= pos );
257  eTime_ = _clock.getTimef();
258 
259  _clock.reset();
260  while( !vector.empty( ))
261  vector.pop_back();
262  const float fTime = _clock.getTimef();
