92  Strings getKeys( const std::string& instance ) const
93  {
94  Strings keys;
95  InstanceMapCIter i = _instanceMap.find( instance );
96  if( i == _instanceMap.end( ))
97  return keys;
98 
99  const ValueMap& values = i->second;
100  for( ValueMapCIter j = values.begin(); j != values.end(); ++j )
101  keys.push_back( j->first );
102  return keys;
103  }

105  bool containsKey( const std::string& instance, const std::string& key )
106  const
107  {
108  InstanceMapCIter i = _instanceMap.find( instance );
109  if( i == _instanceMap.end( ))
110  return false;
111 
112  const ValueMap& values = i->second;
113  ValueMapCIter j = values.find( key );
114  if( j == values.end( ))
115  return false;
116  return true;
117  }
118 
119  const std::string& get( const std::string& instance,
120  const std::string& key ) const
121  {
122  InstanceMapCIter i = _instanceMap.find( instance );
123  if( i == _instanceMap.end( ))
124  return detail::empty_;
125 
126  const ValueMap& values = i->second;
127  ValueMapCIter j = values.find( key );
128  if( j == values.end( ))
129  return detail::empty_;
130  return j->second;
131  }

203  case kDNSServiceErr_NoMemory: return "out of memory";
204  case kDNSServiceErr_BadParam: return "bad parameter";
205  case kDNSServiceErr_BadReference: return "bad reference";
206  case kDNSServiceErr_BadState: return "bad state";
207  case kDNSServiceErr_BadFlags: return "bad flags";
208  case kDNSServiceErr_Unsupported: return "unsupported";
209  case kDNSServiceErr_NotInitialized: return "not initialized";
210  case kDNSServiceErr_AlreadyRegistered: return "already registered";
211  case kDNSServiceErr_NameConflict: return "name conflict";
212  case kDNSServiceErr_Invalid: return "invalid value";
213  case kDNSServiceErr_Firewall: return "firewall";
214  case kDNSServiceErr_Incompatible:
215  return "client library incompatible with daemon";
216  case kDNSServiceErr_BadInterfaceIndex: return "bad interface index";
217  case kDNSServiceErr_Refused: return "refused";
218  case kDNSServiceErr_NoSuchRecord: return "no such record";
219  case kDNSServiceErr_NoAuth: return "no authentication";
220  case kDNSServiceErr_NoSuchKey: return "no such key";
221  case kDNSServiceErr_NATTraversal: return "NAT traversal";
222  case kDNSServiceErr_DoubleNAT: return "double NAT";
223  case kDNSServiceErr_BadTime: return "bad time";
224 #endif
225 
226  case PENDING: return "operation pending";
227  case NOT_SUPPORTED: return "Lunchbox compiled without ZeroConf support";
228  case POLL_ERROR: return "Error polling for events";
229  default:
