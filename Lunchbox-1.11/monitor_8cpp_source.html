78 class Thread2 : public lunchbox::Thread
79 {
80 public:
81  Thread2( size_t loops )
82  : innerLoops( 0 )
83  , _outerLoops( loops )
84  {}
85 
86  virtual ~Thread2() {}
87  virtual void run()
88  {
89  lunchbox::Clock clock;
90  size_t ops = 0;
91  for( size_t k = 0; k != 2; ++k )
92  {
93  for( size_t i = 0; i != _outerLoops; ++i )
94  {
95  const int64_t loops = innerLoops.waitNE( 0 );
96  innerLoops = 0;
97  for( int64_t j = 0; j != loops; ++j )
98  {
99  if (innerLoops > 0)
100  abort();
101  ++monitor;
102  }
103  ops += loops * 2 + 2;
104  }
105  }
106 
107  for( size_t k = 0; k != 2; ++k )
108  {
109  for( size_t i = 0; i != _outerLoops; ++i )
110  {
111  const int64_t loops = innerLoops.waitNE( 0 );
112  innerLoops = 0;
113  for( int64_t j = 0; j != loops; ++j )
114  {
115  if (innerLoops > 0)
116  abort();
117  --monitor;
118  }
119  ops += loops * 2 + 2;
120  }
121  }
122 
123  const float time = clock.getTimef();
124  std::cout << ops/time << " ops/ms" << std::endl;
125  }
