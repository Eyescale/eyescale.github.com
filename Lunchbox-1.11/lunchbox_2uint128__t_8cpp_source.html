48 uint128_t& uint128_t::operator = ( const std::string& from )
49 {
50  if( from.empty( ))
51  {
52  _high = 0;
53  _low = 0;
54  return *this;
55  }
56 
57  char* next = 0;
58  _high = ::strtoull( from.c_str(), &next, 16 );
59  LBASSERT( next != from.c_str( ));
60 
61  if( *next == '\0' ) // short representation, high was 0
62  {
63  _low = _high;
64  _high = 0;
65  }
66  else
67  {
68  if( strncmp( next, "\\058" /* utf-8 ':' */, 4 ) == 0 )
69  next += 4;
70  else
71  {
72  LBASSERTINFO( *next == ':', from << ", " << next );
73  ++next;
74  }
75  _low = ::strtoull( next, 0, 16 );
76  }
77  return *this;
78 }
