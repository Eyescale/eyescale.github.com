90  lunchbox::Servus::Result announce( const unsigned short port,
91  const std::string& instance ) final
92  {
93  ScopedWrite mutex( lock_ );
94  _result = lunchbox::Servus::Result::PENDING;
95  _port = port;
96  if( instance.empty( ))
97  _announce = getHostname();
98  else
99  _announce = instance;
100 
101  if( _announcable )
102  _createServices();
103  else
104  {
105  lunchbox::Clock clock;
106  while( !_announcable &&
107  _result == lunchbox::Servus::Result::PENDING &&
108  clock.getTime64() < ANNOUNCE_TIMEOUT )
109  {
110  avahi_simple_poll_iterate( _poll, ANNOUNCE_TIMEOUT );
111  }
112  }
113 
114  return lunchbox::Servus::Result( _result );
115  }

155  lunchbox::Servus::Result browse( const int32_t timeout ) final
156  {
157  ScopedWrite mutex( lock_ );
158  _result = lunchbox::Servus::Result::PENDING;
159  lunchbox::Clock clock;
160 
161  do
162  {
163  if( avahi_simple_poll_iterate( _poll, timeout ) != 0 )
164  {
165  _result = lunchbox::Servus::Result::POLL_ERROR;
166  break;
167  }
168  }
169  while( clock.getTime64() < timeout );
170 
171  if( _result != lunchbox::Servus::Result::POLL_ERROR )
172  _result = lunchbox::Servus::Result::SUCCESS;
173 
174  return lunchbox::Servus::Result( _result );
175  }

219  void _clientCB( AvahiClientState state )
220  {
221  switch (state)
222  {
223  case AVAHI_CLIENT_S_RUNNING:
224  _announcable = true;
225  if( !_announce.empty( ))
226  _createServices();
227  break;
228 
229  case AVAHI_CLIENT_FAILURE:
230  _result = avahi_client_errno( _client );
231  LBWARN << "Client failure: " << avahi_strerror( _result )
232  << std::endl;
233  avahi_simple_poll_quit( _poll );
234  break;
235 
236  case AVAHI_CLIENT_S_COLLISION:
237  // Can't setup client
238  _result = EEXIST;
239  avahi_simple_poll_quit( _poll );
240  break;
241 
242  case AVAHI_CLIENT_S_REGISTERING:
243  /* The server records are now being established. This might be
244  * caused by a host name change. We need to wait for our own records
245  * to register until the host name is properly esatblished. */
246  LBUNIMPLEMENTED; // withdraw & _createServices ?
247  break;
248 
249  case AVAHI_CLIENT_CONNECTING:
250  /*nop*/;
251  }
252  }

265  void _browseCB( const AvahiIfIndex ifIndex, const AvahiProtocol protocol,
266  const AvahiBrowserEvent event, const char* name,
267  const char* type, const char* domain )
268  {
269  LBVERB << "Browse event " << int(event) << " for "
270  << (name ? name : "none") << " type " << (type ? type : "none")
271  << std::endl;
272  switch( event )
273  {
274  case AVAHI_BROWSER_FAILURE:
275  _result = avahi_client_errno( _client );
276  LBWARN << "Browser failure: " << avahi_strerror( _result )
277  << std::endl;
278  avahi_simple_poll_quit( _poll );
279  break;
280 
281  case AVAHI_BROWSER_NEW:
282  /* We ignore the returned resolver object. In the callback function
283  we free it. If the server is terminated before the callback
284  function is called the server will free the resolver for us. */
285  if( !avahi_service_resolver_new( _client, ifIndex, protocol, name,
286  type, domain, AVAHI_PROTO_UNSPEC,
287  (AvahiLookupFlags)(0),
288  _resolveCBS, this ))
289  {
290  _result = avahi_client_errno( _client );
291  LBWARN << "Error creating resolver: "
292  << avahi_strerror( _result ) << std::endl;
293  avahi_simple_poll_quit( _poll );
294  break;
295  }
296 
297  case AVAHI_BROWSER_REMOVE:
298  _instanceMap.erase( name );
299  break;
300 
301  case AVAHI_BROWSER_ALL_FOR_NOW:
302  case AVAHI_BROWSER_CACHE_EXHAUSTED:
303  _result = lunchbox::Result::SUCCESS;
304  break;
305  }
306  }

320  void _resolveCB( AvahiServiceResolver* resolver,
321  const AvahiResolverEvent event, const char* name,
322  const char* host, AvahiStringList *txt )
323  {
324  // If browsing through the local interface,
325  // consider only the local instances
326  if( _scope == lunchbox::Servus::IF_LOCAL )
327  {
328  const std::string& hostStr( host );
329  // host in "hostname.local" format
330  const size_t pos = hostStr.find_last_of( "." );
331  const std::string hostName = hostStr.substr( 0, pos );
332 
333  const std::string& localHost = getHostname();
334  // omit the domain for the local hostname
335  if( hostName != localHost.substr( 0,
336  localHost.find_first_of( "." )))
337  return;
338  }
339 
340  switch( event )
341  {
342  case AVAHI_RESOLVER_FAILURE:
343  _result = avahi_client_errno( _client );
344  LBWARN << "Resolver error: " << avahi_strerror( _result )
345  << std::endl;
346  break;
347 
348  case AVAHI_RESOLVER_FOUND:
349  {
350  detail::ValueMap& values = _instanceMap[ name ];
351  values[ "servus_host" ] = host;
352  for( ; txt; txt = txt->next )
353  {
354  const std::string entry(
355  reinterpret_cast< const char* >( txt->text ),
356  txt->size );
357  const size_t pos = entry.find_first_of( "=" );
358  const std::string key = entry.substr( 0, pos );
359  const std::string value = entry.substr( pos + 1 );
360  values[ key ] = value;
361  }
362  } break;
363  }
364 
365  avahi_service_resolver_free( resolver );
366  }

379  void _createServices()
380  {
381  if( !_group )
382  _group = avahi_entry_group_new( _client, _groupCBS, this );
383  else
384  avahi_entry_group_reset( _group );
385 
386  if( !_group )
387  return;
388 
389  AvahiStringList* data = 0;
390  for( detail::ValueMapCIter i = _data.begin(); i != _data.end(); ++i )
391  data = avahi_string_list_add_pair( data, i->first.c_str(),
392  i->second.c_str( ));
393 
394  _result = avahi_entry_group_add_service_strlst(
395  _group, AVAHI_IF_UNSPEC, AVAHI_PROTO_UNSPEC,
396  (AvahiPublishFlags)(0), _announce.c_str(), _name.c_str(), 0, 0,
397  _port, data );
398 
399  if( data )
400  avahi_string_list_free( data );
401 
402  if( _result != lunchbox::Result::SUCCESS )
403  {
404  avahi_simple_poll_quit( _poll );
405  return;
406  }
407 
408  _result = avahi_entry_group_commit( _group );
409  if( _result != lunchbox::Result::SUCCESS )
410  avahi_simple_poll_quit( _poll );
411  }
