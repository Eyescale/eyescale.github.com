88 void Clock::set( const int64_t time )
89 {
90  reset();
91 #ifdef __APPLE__
92  _impl->start -= static_cast< uint64_t >(
93  time * _impl->timebaseInfo.denom / _impl->timebaseInfo.numer *
94  1000000 );
95 #elif defined (_WIN32)
96  _impl->start.QuadPart -= static_cast<long long>(
97  time * _impl->frequency.QuadPart / 1000 );
98 #else
99  const int sec = static_cast< int >( time / 1000 ) + 1;
100  _impl->start.tv_sec -= sec;
101  _impl->start.tv_nsec -= static_cast<int>(
102  (time - sec * 1000) * 1000000 );
103  if( _impl->start.tv_nsec > 1000000000 )
104  {
105  _impl->start.tv_sec += 1;
106  _impl->start.tv_nsec -= 1000000000;
107  }
108 #endif
109 }

111 float Clock::getTimef() const
112 {
113 #ifdef __APPLE__
114  const int64_t elapsed = mach_absolute_time() - _impl->start;
115  return ( elapsed * _impl->timebaseInfo.numer / _impl->timebaseInfo.denom /
116  1000000.f );
117 #elif defined (_WIN32)
118  LARGE_INTEGER now;
119  QueryPerformanceCounter( &now );
120  return 1000.0f * (now.QuadPart - _impl->start.QuadPart) /
121  _impl->frequency.QuadPart;
122 #else
123  struct timespec now;
124  clock_gettime( CLOCK_REALTIME, &now );
125  return ( 1000.0f * (now.tv_sec - _impl->start.tv_sec) +
126  0.000001f * (now.tv_nsec - _impl->start.tv_nsec));
127 #endif
128 }

190 float Clock::getMilliSecondsf() const
191 {
192 #if defined (__APPLE__) || defined (_WIN32)
193  double time = getTimed();
194  return static_cast<float>
195  (time - static_cast<unsigned>(time/1000.) * 1000);
196 #else
197  struct timespec now;
198  clock_gettime( CLOCK_REALTIME, &now );
199 
200  if( now.tv_nsec < _impl->start.tv_nsec )
201  return ( 1000.f + 0.000001f*(now.tv_nsec - _impl->start.tv_nsec));
202 
203  return ( 0.000001f * ( now.tv_nsec - _impl->start.tv_nsec ));
204 #endif
