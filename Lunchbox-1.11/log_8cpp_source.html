86  void setLogInfo( const char* f, const int line )
87  {
88  LBASSERT( f );
89  std::string file( f );
90  const size_t length = file.length();
91 
92  if( length > 29 )
93  file = file.substr( length - 29, length );
94 
95  snprintf( _file, 35, "%29s:%-4d", file.c_str(), line );
96  }

98 protected:
99  virtual int_type overflow( Log::int_type c ) override
100  {
101  if( c == EOF )
102  return EOF;
103 
104  if( _newLine )
105  {
106  if( !_noHeader )
107  {
108  _stringStream << getpid() << "." << _thread << " " << _file
109  << " " << _clock->getTime64() << " ";
110  }
111 
112  for( int i=0; i<_indent; ++i )
113  _stringStream << " ";
114  _newLine = false;
115  }
116 
117  _stringStream << (char)c;
118  return c;
119  }

121  virtual int sync() override
122  {
123  if( !_blocked )
124  {
125  const std::string& string = _stringStream.str();
126  {
127  ScopedMutex< lunchbox::Lock > mutex( _lock );
128  _stream.write( string.c_str(), string.length( ));
129  _stream.rdbuf()->pubsync();
130  }
131  _stringStream.str( "" );
132  }
133  _newLine = true;
134  return 0;
135  }
