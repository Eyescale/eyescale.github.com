134 inline std::ostream& operator << ( std::ostream& os, const URI& uri )
135 {
136  if( !uri.getScheme().empty( ))
137  os << uri.getScheme() << "://";
138  if( !uri.getUserinfo().empty( ))
139  os << uri.getUserinfo() << "@";
140  os << uri.getHost();
141  if( uri.getPort( ))
142  os << ':' << uri.getPort();
143  os << uri.getPath();
144  if( !uri.getQuery().empty( ))
145  os << '?' << uri.getQuery();
146  if( !uri.getFragment().empty( ))
147  os << '#' << uri.getFragment();
148  return os;
