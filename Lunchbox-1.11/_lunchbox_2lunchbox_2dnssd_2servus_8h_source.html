190  lunchbox::Servus::Result _handleEvents( DNSServiceRef service,
191  const int32_t timeout = -1 )
192  {
193  LBASSERT( service );
194  if( !service )
195  return lunchbox::Servus::Result( kDNSServiceErr_Unknown );
196 
197  const int fd = DNSServiceRefSockFD( service );
198  const int nfds = fd + 1;
199 
200  LBASSERT( fd >= 0 );
201  if( fd < 0 )
202  return lunchbox::Servus::Result( kDNSServiceErr_BadParam );
203 
204  while( _result == lunchbox::Servus::Result::PENDING )
205  {
206  fd_set fdSet;
207  FD_ZERO( &fdSet );
208  FD_SET( fd, &fdSet );
209 
210  struct timeval tv;
211  tv.tv_sec = timeout / 1000;
212  tv.tv_usec = (timeout % 1000) * 1000;
213 
214  const int result = ::select( nfds, &fdSet, 0, 0,
215  timeout < 0 ? 0 : &tv );
216  switch( result )
217  {
218  case 0: // timeout
219  _result = kDNSServiceErr_NoError;
220  break;
221 
222  case -1: // error
223  LBWARN << "Select error: " << strerror( errno ) << " (" << errno
224  << ")" << std::endl;
225  if( errno != EINTR )
226  {
227  withdraw();
228  _result = errno;
229  }
230  break;
231 
232  default:
233  if( FD_ISSET( fd, &fdSet ))
234  {
235  const DNSServiceErrorType error =
236  DNSServiceProcessResult( service );
237 
238  if( error != kDNSServiceErr_NoError )
239  {
240  LBWARN << "DNSServiceProcessResult error: " << error
241  << std::endl;
242  withdraw();
243  _result = error;
244  }
245  }
246  break;
247  }
248  }
249 
250  const lunchbox::Servus::Result result( _result );
251  _result = lunchbox::Servus::Result::PENDING; // reset for next operation
252  return result;
253  }

285  void browseCB_( DNSServiceFlags flags, uint32_t interfaceIdx,
286  DNSServiceErrorType error, const char* name,
287  const char* type, const char* domain )
288  {
289  if( error != kDNSServiceErr_NoError)
290  {
291  LBWARN << "Browse callback error: " << error << std::endl;
292  return;
293  }
294 
295  if( flags & kDNSServiceFlagsAdd )
296  {
297  _browsedName = name;
298 
299  DNSServiceRef service = 0;
300  const DNSServiceErrorType resolve =
301  DNSServiceResolve( &service, 0, interfaceIdx, name, type,
302  domain, (DNSServiceResolveReply)resolveCBS_,
303  this );
304  if( resolve != kDNSServiceErr_NoError)
305  LBWARN << "DNSServiceResolve error: " << resolve << std::endl;
306 
307  if( service )
308  {
309  _handleEvents( service, 500 );
310  DNSServiceRefDeallocate( service );
311  }
312  }
313  else // dns_sd.h: callback with the Add flag NOT set indicates a Remove
314  _instanceMap.erase( name );
315  }

331  const unsigned char* txt )
332  {
333  detail::ValueMap& values = _instanceMap[ _browsedName ];
334  values[ "servus_host" ] = host;
335 
336  char key[256] = {0};

339 
340  uint16_t i = 0;

342  &valueLen, (const void**)( &value )) ==
343  kDNSServiceErr_NoError )
