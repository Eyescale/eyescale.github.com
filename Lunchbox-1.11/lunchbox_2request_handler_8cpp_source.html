60 class RequestHandler
61 {
62 public:
63  RequestHandler() : requestID( 1 ) {}
64 
65  uint32_t registerRequest( void* data )
66  {
67  ScopedFastWrite mutex( lock );
68 
69  Record* request;
70  if( freeRecords.empty( ))
71  request = new Record;
72  else
73  {
74  request = freeRecords.front();
75  freeRecords.pop_front();
76  }
77 
78  request->data = data;
79  requestID = ( requestID + 1 ) % LB_MAX_UINT32;
80  requests[ requestID ] = request;
81  return requestID;
82  }
83 
84  bool waitRequest( const uint32_t requestID_, Record::Result& result,
85  const uint32_t timeout )
86  {
87  result.rUint128.low = 0;
88  result.rUint128.high = 0;
89  Record* request = 0;
90  {
91  ScopedFastWrite mutex( lock );
92  RecordHashCIter i = requests.find( requestID_ );
93  if( i == requests.end( ))
94  return false;
95 
96  request = i->second;
97  }
98 
99  const bool requestServed = request->lock.set( timeout );
100  if( requestServed )
101  result = request->result;
102 
103  unregisterRequest( requestID_ );
104  return requestServed;
105  }
106 
107  void unregisterRequest( const uint32_t requestID_ )
108  {
109  ScopedFastWrite mutex( lock );
110  RecordHash::iterator i = requests.find( requestID_ );
111  if( i == requests.end( ))
112  return;
113 
114  Record* request = i->second;
115  requests.erase( i );
116  freeRecords.push_front( request );
117  }
118 
119  mutable lunchbox::SpinLock lock;
120  uint32_t requestID;
121  RecordHash requests;
122  std::list<Record*> freeRecords;
123 };
124 }
