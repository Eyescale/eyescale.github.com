36 bool init( const int argc, char** argv )
37 {
38 #ifndef NDEBUG
39  LBVERB << "Options: ";
40  for( int i = 1; i < argc; ++i )
41  LBVERB << argv[i] << ", ";
42  LBVERB << std::endl;
43 #endif
44 
45  for( int i = 1; i < argc; ++i )
46  {
47  // verbose options
48  if( std::string( argv[i] ) == "-vv" )
49  Log::level = LOG_VERB;
50  else if( std::string( argv[i] ) == "-v" )
51  Log::level = LOG_INFO;
52  }
53 
54  if( ++_initialized > 1 ) // not first
55  return true;
56 
57  Log::instance().setThreadName( "Main" );
58 
59  const time_t now = ::time(0);
60 #ifdef _WIN32
61  char* gmtString = ::ctime( &now );
62 #else
63  char gmtString[32];
64  ::ctime_r( &now, gmtString );
65 
66  setenv( "AVAHI_COMPAT_NOWARN", "1", 0 ); // get rid of annoying avahi warning
67 #endif
68 
69  LBINFO << "Log level " << Log::getLogLevelString() << " topics "
70  << Log::topics << " date " << gmtString << std::flush;
71 
72  if( !RNG::_init( ))
73  {
74  LBERROR << "Failed to initialize random number generator" << std::endl;
75  return false;
76  }
77 
78  Thread::pinCurrentThread();
79  return true;
80 }
