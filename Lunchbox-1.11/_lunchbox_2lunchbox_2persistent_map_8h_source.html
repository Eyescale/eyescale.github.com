225  template< class V > V _get( const std::string& k ) const
226  {
227  if( !boost::has_trivial_assign< V >( ))
228  LBTHROW( std::runtime_error( "Can't retrieve non-POD " +
229  className( V( ))));
230  if( boost::is_pointer< V >::value )
231  LBTHROW( std::runtime_error( "Can't retrieve pointers" ));
232 
233  const std::string& value = (*this)[ k ];
234  if( value.size() != sizeof( V ))
235  LBTHROW( std::runtime_error( std::string( "Wrong value size " ) +
236  boost::lexical_cast< std::string >( value.size( )) +
237  " for type " + className( V( ))));
238 
239  V v( *reinterpret_cast< const V* >( value.data( )));
240  if( _swap( ))
241  byteswap( v );
242  return v;
243  }

253 template< class V > inline
254 std::vector< V > PersistentMap::getVector( const std::string& key ) const
255 {
256  const std::string& value = (*this)[ key ];
257  std::vector< V > vector( reinterpret_cast< const V* >( value.data( )),
258  reinterpret_cast< const V* >( value.data() + value.size( )));
259  if( _swap() && sizeof( V ) != 1 )
260  {
261  BOOST_FOREACH( V& elem, vector )
262  byteswap( elem );
263  }
264  return vector;

267 template< class V > inline
268 std::set< V > PersistentMap::getSet( const std::string& key ) const
269 {
270  std::string value = (*this)[ key ];
271  V* const begin = reinterpret_cast< V* >(
272  const_cast< char * >( value.data( )));
273  V* const end = begin + value.size() / sizeof( V );
274 
275  if( _swap() && sizeof( V ) != 1 )
276  for( V* i = begin; i < end; ++i )
277  byteswap( *i );
278 
279  return std::set< V >( begin, end );
