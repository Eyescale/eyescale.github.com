82 class URI
83 {
84 public:
85  explicit URI( const std::string& uri )
86  {
87  if( uri.empty( ))
88  return;
89 
90  boost::match_results< std::string::const_iterator > results;
91  boost::regex expr(
92  "^(([^:/?#]+):)?(//([^/?#]*))?([^?#]*)(\\?([^#]*))?(#([^?]*))?$",
93  boost::regex::perl | boost::regex::icase );
94 
95  if( !boost::regex_search( uri, results, expr ) )
96  throw uri_parse( uri );
97 
98  _uriData.scheme = std::string( results[2].first, results[2].second );
99 
100  const std::string& userHost = std::string( results[4].first,
101  results[4].second );
102  if( !userHost.empty( ))
103  {
104  std::vector< std::string > splitUserHost;
105  std::string hostPort;
106  boost::algorithm::split( splitUserHost, userHost,
107  boost::is_any_of( "@"));
108  if( splitUserHost.size() == 2 ) // for ex: user:pass@hello.com:port
109  {
110  _uriData.userinfo = splitUserHost[ 0 ];
111  hostPort = splitUserHost[ 1 ];
112  }
113  else
114  hostPort = splitUserHost[ 0 ];
115 
116  std::vector< std::string > splitHostPort;
117  boost::algorithm::split( splitHostPort, hostPort,
118  boost::is_any_of( ":" ));
119  _uriData.host = splitHostPort[ 0 ];
120 
121  if( splitHostPort.size() == 2 ) // for ex: myhost:port
122  _uriData.port = boost::lexical_cast< uint16_t >(
123  splitHostPort[ 1 ] );
124  }
125 
126  _uriData.path = std::string( results[5].first, results[5].second );
127  _uriData.query = std::string( results[7].first, results[7].second );
128  _uriData.fragment = std::string( results[9].first, results[9].second );
129 
130  // from http://en.wikipedia.org/wiki/File_URI_scheme:
131  // "file:///foo.txt" is okay, while "file://foo.txt" is not, although
132  // some interpreters manage to handle the latter We are "some".
133  const bool isFileURI = _uriData.scheme.empty() ||
134  _uriData.scheme == "file";
135  const bool hasHost = !_uriData.host.empty();
136  const bool hasPath = !_uriData.path.empty();
137  if( isFileURI && hasHost && !hasPath )
138  _uriData.host.swap( _uriData.path );
139 
140  // parse query data into key-value pairs
141  std::string query = _uriData.query;
142  while( !query.empty( ))
143  {
144  const size_t nextPair = query.find( ',' );
145  if( nextPair == 0 )
146  {
147  query = query.substr( 1 );
148  continue;
149  }
150 
151  const std::string pair = query.substr( 0, nextPair );
152  if( nextPair == std::string::npos )
153  query.clear();
154  else
155  query = query.substr( nextPair + 1 );
156 
157  const size_t eq = pair.find( '=' );
158  if( eq == std::string::npos || eq == 0 )
159  continue;
160  _uriData.queryMap[ pair.substr( 0, eq ) ] = pair.substr( eq + 1 );
161  }
162  }
163 
164  URIData& getData() { return _uriData; }
165  const URIData& getData() const { return _uriData; }
166 
167 private:
168  URIData _uriData;
169 };

269 void URI::addQuery( const std::string& key, const std::string& value )
270 {
271  URIData& data = _impl->getData();
272 
273  data.queryMap[ key ] = value;
274  data.fragment.clear();
275 
276  // Rebuild fragment string
277  data.query.clear();
278  BOOST_FOREACH( const URI::KVMap::value_type& pair, data.queryMap )
279  {
280  if( data.query.empty( ))
281  data.query = pair.first + "=" + pair.second;
282  else
283  data.query += std::string( "," ) + pair.first + "=" + pair.second;
284  }
