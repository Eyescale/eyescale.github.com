114 void Thread::_runChild()
115 {
116  setName( boost::lexical_cast< std::string >( _impl->index ));
117  pinCurrentThread();
118  _impl->id._impl->pthread = pthread_self();
119 
120  if( !init( ))
121  {
122  LBWARN << "Thread " << className( this ) << " failed to initialize"
123  << std::endl;
124  _impl->state = STATE_STOPPED;
125  pthread_exit( 0 );
126  LBUNREACHABLE;
127  }
128 
129  _impl->state = STATE_RUNNING;
130  LBINFO << "Thread #" << _impl->index << " type " << className( *this )
131  << " successfully initialized" << std::endl;
132 
133  run();
134  LBVERB << "Thread " << className( this ) << " finished" << std::endl;
135  this->exit();
136 
137  LBUNREACHABLE;
138 }

140 bool Thread::start()
141 {
142  if( _impl->state != STATE_STOPPED )
143  return false;
144 
145  _impl->state = STATE_STARTING;
146 
147  pthread_attr_t attributes;
148  pthread_attr_init( &attributes );
149  pthread_attr_setscope( &attributes, PTHREAD_SCOPE_SYSTEM );
150 
151  int nTries = 10;
152  while( nTries-- )
153  {
154  const int error = pthread_create( &_impl->id._impl->pthread,
155  &attributes, runChild, this );
156  if( error == 0 ) // succeeded
157  {
158  LBVERB << "Created pthread " << this << std::endl;
159  break;
160  }
161  if( error != EAGAIN || nTries == 0 )
162  {
163  LBWARN << "Could not create thread: " << strerror( error )
164  << std::endl;
165  return false;
166  }
167  sleep( 1 ); // Give EAGAIN some time to recover
168  }
169 
170  // avoid memleak, we don't use pthread_join
171  pthread_detach( _impl->id._impl->pthread );
172  _impl->state.waitNE( STATE_STARTING );
173  return (_impl->state != STATE_STOPPED);
174 }

238 void Thread::pinCurrentThread()
239 {
240 #ifdef LB_WIN32_THREAD_AFFINITY
241  static Lock lock;
242  ScopedMutex<> mutex( lock );
243 
244  static DWORD_PTR processMask = 0;
245  static DWORD_PTR processor = 0;
246  if( processMask == 0 )
247  {
248  // Get available processors
249  DWORD_PTR systemMask;
250  if( GetProcessAffinityMask( GetCurrentProcess(), &processMask,
251  &systemMask ) == 0 )
252  {
253  LBWARN << "Can't get usable processor mask" << std::endl;
254  return;
255  }
256  LBINFO << "Available processors 0x" << hex << processMask << dec <<endl;
257 
258  // Choose random starting processor: Multiple Eq apps on the same node
259  // would otherwise use the same processor for the same thread
260  unsigned nProcessors = 0;
261  for( DWORD_PTR i = 1; i != 0; i <<= 1 )
262  {
263  if( processMask & i )
264  ++nProcessors;
265  }
266  LBINFO << nProcessors << " available processors" << std::endl;
267 
268  unsigned chance = RNG().get< unsigned >();
269  processor = 1 << (chance % nProcessors);
270  LBINFO << "Starting with processor " << processor << std::endl;
271  }
272  LBASSERT( processMask != 0 );
273 
274  while( true )
275  {
276  processor <<= 1;
277  if( processor == 0 ) // wrap around
278  processor = 1;
279 
280  if( processor & processMask ) // processor is available
281  {
282  if( SetThreadAffinityMask( GetCurrentThread(), processor ) == 0 )
283  LBWARN << "Can't set thread processor" << std::endl;
284  LBINFO << "Pinned thread to processor 0x" << hex << processor << dec
285  << std::endl;
286  return;
287  }
288  }
289 #endif
290 }

345 #ifdef LUNCHBOX_USE_HWLOC
346 static hwloc_bitmap_t _getCpuSet( const int32_t affinity,
347  hwloc_topology_t topology )
348 {
349  hwloc_bitmap_t cpuSet = hwloc_bitmap_alloc(); // HWloc CPU set
350  hwloc_bitmap_zero( cpuSet ); // Initialize to zeros
351 
352  if( affinity >= Thread::CORE )
353  {
354  const int32_t coreIndex = affinity - Thread::CORE;
355  if( hwloc_get_obj_by_type( topology, HWLOC_OBJ_CORE, coreIndex ) == 0 )
356  {
357  LBWARN << "Core " << coreIndex << " does not exist in the topology"
358  << std::endl;
359  return cpuSet;
360  }
361 
362  // Getting the core object #coreIndex
363  const hwloc_obj_t coreObj = hwloc_get_obj_by_type( topology,
364  HWLOC_OBJ_CORE,
365  coreIndex );
366  // Get the CPU set associated with the specified core
367  cpuSet = coreObj->allowed_cpuset;
368  return cpuSet;
369  }
370 
371  if( affinity == Thread::NONE )
372  return cpuSet;
373 
374  // Sets the affinity to a specific CPU or "socket"
375  LBASSERT( affinity >= Thread::SOCKET && affinity < Thread::SOCKET_MAX );
376  const int32_t socketIndex = affinity - Thread::SOCKET;
377 
378  if( hwloc_get_obj_by_type( topology, HWLOC_OBJ_SOCKET, socketIndex ) == 0 )
379  {
380  LBWARN << "Socket " << socketIndex << " does not exist in the topology"
381  << std::endl;
382  return cpuSet;
383  }
384 
385  // Getting the CPU object #cpuIndex (subtree node)
386  const hwloc_obj_t socketObj = hwloc_get_obj_by_type( topology,
387  HWLOC_OBJ_SOCKET,
388  socketIndex );
389  // Get the CPU set associated with the specified socket
390  hwloc_bitmap_copy( cpuSet, socketObj->allowed_cpuset );
391  return cpuSet;
392 }
393 #endif

395 void Thread::setAffinity( const int32_t affinity )
396 {
397  if( affinity == Thread::NONE )
398  return;
399 
400 #ifdef LUNCHBOX_USE_HWLOC
401  hwloc_topology_t topology;
402  hwloc_topology_init( &topology ); // Allocate & initialize the topology
403  hwloc_topology_load( topology ); // Perform HW topology detection
404  const hwloc_bitmap_t cpuSet = _getCpuSet( affinity, topology );
405  const int result = hwloc_set_cpubind( topology, cpuSet,
406  HWLOC_CPUBIND_THREAD );
407  char* cpuSetString;
408  hwloc_bitmap_asprintf( &cpuSetString, cpuSet );
409 
410  if( result == 0 )
411  {
412  LBVERB << "Bound to cpu set " << cpuSetString << std::endl;
413  }
414  else
415  {
416  LBWARN << "Error binding to cpu set " << cpuSetString << std::endl;
417  }
418  ::free( cpuSetString );
419  hwloc_bitmap_free( cpuSet );
420  hwloc_topology_destroy( topology );
421 
422 #else
423  LBWARN << "Thread::setAffinity not implemented, hwloc library missing"
424  << std::endl;
425 #endif
426 }
