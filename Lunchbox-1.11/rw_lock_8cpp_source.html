105 template< class T, uint32_t hold > void _test()
106 {
107  T* lock = new T;
108  lock->set();
109 
110 #ifdef LUNCHBOX_USE_OPENMP
111  const size_t nThreads = LB_MIN( lunchbox::OMP::getNThreads()*3, MAXTHREADS );
112 #else
113  const size_t nThreads = 16;
114 #endif
115 
116  WriteThread< T, hold > writers[MAXTHREADS];
117  ReadThread< T, hold > readers[MAXTHREADS];
118 
119  std::cout << " Class, write ops/ms, read ops/ms, w threads, "
120  << "r threads" << std::endl;
121  for( size_t nWrite = 0; nWrite <= nThreads;
122  nWrite = (nWrite == 0) ? 1 : nWrite << 1 )
123  {
124  for( size_t i = 1; i <= nThreads; i = i << 1 )
125  {
126  if( i < nWrite )
127  continue;
128 
129  const size_t nRead = i - nWrite;
130  _running = true;
131  for( size_t j = 0; j < nWrite; ++j )
132  {
133  writers[j].lock = lock;
134  TEST( writers[j].start( ));
135  }
136  for( size_t j = 0; j < nRead; ++j )
137  {
138  readers[j].lock = lock;
139  TESTINFO( readers[j].start(), j );
140  }
141  lunchbox::sleep( 10 ); // let threads initialize
142 
143  _clock.reset();
144  lock->unset();
145  lunchbox::sleep( TIME ); // let threads run
146  _running = false;
147 
148  for( size_t j = 0; j < nWrite; ++j )
149  TEST( writers[j].join( ));
150  for( size_t j = 0; j < nRead; ++j )
151  TEST( readers[j].join( ));
152  const double time = _clock.getTimed();
153 
154  TEST( !lock->isSet( ));
155  lock->set();
156 
157  size_t nWriteOps = 0;
158  double wTime = time * double( nWrite );
159  for( size_t j = 0; j < nWrite; ++j )
160  {
161  nWriteOps += writers[j].ops;
162  wTime -= writers[j].sTime;
163  }
164  if( nWrite > 0 )
165  wTime /= double( nWrite );
166  if( wTime == 0.f )
167  wTime = std::numeric_limits< double >::epsilon();
168 
169 
170  size_t nReadOps = 0;
171  double rTime = time * double( nRead );
172  for( size_t j = 0; j < nRead; ++j )
173  {
174  nReadOps += readers[j].ops;
175  rTime -= readers[j].sTime;
176  }
177  if( nRead > 0 )
178  rTime /= double( nRead );
179  if( rTime == 0.f )
180  rTime = std::numeric_limits< double >::epsilon();
181 
182  std::cout << std::setw(20)<< lunchbox::className( lock ) << ", "
183  << std::setw(12) << 3 * nWriteOps / wTime << ", "
184  << std::setw(12) << 3 * nReadOps / rTime << ", "
185  << std::setw(9) << nWrite << ", " << std::setw(9) << nRead
186  << std::endl;
187  }
188  }
189 
190  delete lock;
191 }
