154 void benchmark( const std::string& uri, const uint64_t queueDepth )
155 {
156  PersistentMap map( uri );
157  map.setQueueDepth( queueDepth );
158 
159  // Prepare keys
160  lunchbox::Strings keys;
161  keys.resize( queueDepth + 1 );
162  for( uint64_t i = 0; i <= queueDepth; ++i )
163  keys[i].assign( reinterpret_cast< char* >( &i ), 8 );
164 
165  // write performance
166  lunchbox::Clock clock;
167  uint64_t i = 0;
168  while( clock.getTime64() < loopTime )
169  {
170  std::string& key = keys[ i % (queueDepth+1) ];
171  *reinterpret_cast< uint64_t* >( &key[0] ) = i;
172  map.insert( key, key );
173  ++i;
174  }
175  map.flush();
176  const float insertTime = clock.getTimef();
177  const uint64_t wOps = i;
178  TEST( i > queueDepth );
179 
180  // read performance
181  std::string key;
182  key.assign( reinterpret_cast< char* >( &i ), 8 );
183 
184  clock.reset();
185  if( queueDepth == 0 ) // sync read
186  {
187  for( i = 0; i < wOps && clock.getTime64() < loopTime; ++i ) // read keys
188  {
189  *reinterpret_cast< uint64_t* >( &key[0] ) = i - queueDepth;
190  map[ key ];
191  }
192  }
193  else // fetch + async read
194  {
195  for( i = 0; i < queueDepth; ++i ) // prefetch queueDepth keys
196  {
197  *reinterpret_cast< uint64_t* >( &key[0] ) = i;
198  TEST( map.fetch( key, 8 ) );
199  }
200 
201  for( ; i < wOps && clock.getTime64() < loopTime; ++i ) // read keys
202  {
203  *reinterpret_cast< uint64_t* >( &key[0] ) = i - queueDepth;
204  map[ key ];
205 
206  *reinterpret_cast< uint64_t* >( &key[0] ) = i;
207  TEST( map.fetch( key, 8 ));
208  }
209 
210  for( uint64_t j = i - queueDepth; j <= i; ++j ) // drain fetched keys
211  {
212  *reinterpret_cast< uint64_t* >( &key[0] ) = j;
213  map[ key ];
214  }
215  }
216 
217  const float readTime = clock.getTimef();
218  const size_t rOps = i;
219 
220  // fetch performance
221  clock.reset();
222  for( i = 0; i < wOps && clock.getTime64() < loopTime && i < LB_128KB; ++i )
223  {
224  *reinterpret_cast< uint64_t* >( &key[0] ) = i;
225  TEST( map.fetch( key, 8 ) );
226  }
227  const float fetchTime = clock.getTimef();
228  const size_t fOps = i;
229 
230  std::cout << boost::format( "%7.2f, %7.2f, %7.2f, %6i")
231  // cppcheck-suppress zerodivcond
232  % queueDepth % (rOps/insertTime) % (wOps/readTime) % (fOps/fetchTime)
233  << std::endl;
234 
235  // check contents of store (not all to save time on bigger tests)
236  for( uint64_t j = 0; j < wOps && clock.getTime64() < loopTime; ++j )
237  {
238  *reinterpret_cast< uint64_t* >( &key[0] ) = j;
239  TESTINFO( map.get< uint64_t >( key ) == j,
240  j << " = " << map.get< uint64_t >( key ));
241  }
242 
243  // try to make sure there's nothing outstanding if we messed up i our test.
244  map.flush();
245 }
