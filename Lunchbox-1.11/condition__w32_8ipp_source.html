76 int pthread_cond_init( pthread_cond_t* cond, const pthread_condattr_t* )
77 {
78  InitializeCriticalSection( &cond->waiters_count_lock_ );
79 
80  // Initialize the count to 0.
81  cond->waiters_count_ = 0;
82 
83  // Create an auto-reset event.
84  cond->events_[pthread_cond_t::SIGNAL] =
85  CreateEvent( NULL, // no security
86  FALSE, // auto-reset event
87  FALSE, // non-signaled initially
88  NULL ); // unnamed
89 
90  // Create a manual-reset event.
91  cond->events_[pthread_cond_t::BROADCAST] =
92  CreateEvent( NULL, // no security
93  TRUE, // manual-reset
94  FALSE, // non-signaled initially
95  NULL ); // unnamed
96 
97  return 0;
98 }

108 int pthread_cond_signal( pthread_cond_t* cond )
109 {
110  EnterCriticalSection( &cond->waiters_count_lock_ );
111  int have_waiters = cond->waiters_count_ > 0;
112  LeaveCriticalSection( &cond->waiters_count_lock_ );
113 
114  if( have_waiters )
115  SetEvent( cond->events_[pthread_cond_t::SIGNAL] );
116  return 0;
117 }

130 int pthread_cond_wait( pthread_cond_t* cond, pthread_mutex_t* mutex )
131 {
132  EnterCriticalSection( &cond->waiters_count_lock_ );
133  cond->waiters_count_++;
134  LeaveCriticalSection( &cond->waiters_count_lock_ );
135 
136  // It's ok to release the <external_mutex> here since Win32
137  // manual-reset events maintain state when used with
138  // <SetEvent>. This avoids the "lost wakeup" bug...
139  LeaveCriticalSection( mutex );
140 
141  // Wait for either event to become signaled due to <pthread_cond_signal>
142  // being called or <pthread_cond_broadcast> being called.
143  int result = WaitForMultipleObjects( 2, cond->events_, FALSE, INFINITE);
144 
145  EnterCriticalSection( &cond->waiters_count_lock_ );
146  cond->waiters_count_--;
147  bool last_waiter =
148  result == WAIT_OBJECT_0 + pthread_cond_t::BROADCAST &&
149  cond->waiters_count_ == 0;
150  LeaveCriticalSection( &cond->waiters_count_lock_ );
151 
152  // Some thread called <pthread_cond_broadcast>.
153  if( last_waiter )
154  // We're the last waiter to be notified or to stop waiting, so
155  // reset the manual event.
156  ResetEvent( cond->events_[pthread_cond_t::BROADCAST] );
157 
158  // Reacquire the <external_mutex>.
159  EnterCriticalSection( mutex );
160 
161  return 0;
162 }

164 int pthread_cond_timedwait_w32_np( pthread_cond_t* cond, pthread_mutex_t* mutex,
165  const unsigned timeout )
166 {
167  EnterCriticalSection( &cond->waiters_count_lock_ );
168  cond->waiters_count_++;
169  LeaveCriticalSection( &cond->waiters_count_lock_ );
170 
171  // It's ok to release the <external_mutex> here since Win32
172  // manual-reset events maintain state when used with
173  // <SetEvent>. This avoids the "lost wakeup" bug...
174  LeaveCriticalSection( mutex );
175 
176  // Wait for either event to become signaled due to <pthread_cond_signal>
177  // being called or <pthread_cond_broadcast> being called.
178  int result = WaitForMultipleObjects( 2, cond->events_, FALSE, timeout );
179 
180  EnterCriticalSection( &cond->waiters_count_lock_ );
181  cond->waiters_count_--;
182  bool last_waiter =
183  result == WAIT_OBJECT_0 + pthread_cond_t::BROADCAST &&
184  cond->waiters_count_ == 0;
185  LeaveCriticalSection( &cond->waiters_count_lock_ );
186 
187  // Some thread called <pthread_cond_broadcast>.
188  if( last_waiter )
189  // We're the last waiter to be notified or to stop waiting, so
190  // reset the manual event.
191  ResetEvent( cond->events_[pthread_cond_t::BROADCAST] );
192 
193  // Reacquire the <external_mutex>.
194  EnterCriticalSection( mutex );
195 
196  if ( result == WAIT_TIMEOUT )
197  return ETIMEDOUT;
198  return 0;
199 }
