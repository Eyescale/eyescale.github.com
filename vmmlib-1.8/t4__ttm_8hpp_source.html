144  // TODO can it be done more efficiently?
145  for (size_t i = 0; i < J1; ++i) {
146  for (size_t j = 0; j < J2; ++j) {
147  for (size_t k = 0; k < J3; ++k) {
148  for (size_t newL = 0; newL < I4; ++newL) {
149  T sum = 0;

151  sum += t4_in_.at(i,j,k,l)*in_slice_.at(newL,l);
