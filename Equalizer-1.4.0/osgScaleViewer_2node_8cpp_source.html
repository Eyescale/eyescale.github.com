51 bool Node::configInit( const eq::uint128_t& initID )
52 {
53  if( !eq::Node::configInit( initID ))
54  return false;
55 
56  Config* config = static_cast<Config*>( getConfig( ));
57  if( !config->mapData( initID ))
58  return false;
59 
60  _frameStamp = new osg::FrameStamp;
61  _updateVisitor = new osgUtil::UpdateVisitor;
62  _updateVisitor->setFrameStamp( _frameStamp );
63  _contextID = 0;
64 
65  // load model at first config run
66  if( !_model )
67  {
68  const InitData& initData = config->getInitData();
69  const std::string& modelFile = initData.getModelFileName();
70  if( !modelFile.empty( ))
71  {
72  SceneReader sceneReader;
73  _model = sceneReader.readModel( modelFile );
74 
75  if( _model.valid( ))
76  {
77  osg::Matrix matrix;
78  matrix.makeRotate( -osg::PI_2, osg::Vec3( 1., 0., 0. ));
79 
80  osg::ref_ptr<osg::MatrixTransform> transform =
81  new osg::MatrixTransform();
82  transform->setMatrix( matrix );
83  transform->addChild( _model );
84  transform->setDataVariance( osg::Object::STATIC );
85 
86  _model = transform;
87  }
88  }
89  }
90 
91  if( !_model )
92  {
93  const InitData& initData = config->getInitData();
94  std::string imageFile = initData.getImageFileName();
95  if( !imageFile.empty( ))
96  {
97  SceneReader sceneReader;
98  osg::ref_ptr<osg::Image> image = sceneReader.readImage( imageFile );
99  if( image.valid( ))
100  _model = _createSceneGraph( image );
101  }
102  }
103 
104  if( !_model )
105  _model = _createSceneGraph();
106  return true;
107 }

118 void Node::frameStart( const eq::uint128_t& frameID,

122 
123  // TODO use global time saved in FrameData, use one FrameData per node

125  _frameStamp->setReferenceTime( time );
126  _frameStamp->setSimulationTime( time );
127  _updateVisitor->setTraversalNumber( frameNumber );

129  _model->getBound();
130 
131  eq::Node::frameStart( frameID, frameNumber );
132 }

147 osg::ref_ptr< osg::Node > Node::_createSceneGraph(
148  osg::ref_ptr< osg::Image > image )
149 {
150  // init scene graph
151  osg::ref_ptr<osg::Group> root = _initSceneGraph();
152 
153  // det the image as a texture
154  osg::Texture2D* texture = new osg::Texture2D();
155  texture->setImage( image );
156 
157  // draw a textured quad
158  Quad quad;
159  osg::ref_ptr<osg::Node> geometryChild = quad.createQuad( image->s(),
160  image->t( ));
161 
162  osg::StateSet* stateOne = new osg::StateSet();
163  stateOne->setTextureAttributeAndModes( 0, texture, osg::StateAttribute::ON);
164  geometryChild->setStateSet( stateOne );
165 
166  root->addChild( geometryChild );
167 
168  return root.get();
169 }

171 osg::ref_ptr< osg::Group > Node::_initSceneGraph()
172 {
173  osg::ref_ptr<osg::Group> root = new osg::Group();
174  root->setDataVariance( osg::Object::STATIC );
175 
176  // enable lighting and LIGHT0 in root node
177  osg::StateSet* state = root->getOrCreateStateSet();
178  state->setMode( GL_LIGHTING, osg::StateAttribute::ON );
179  state->setMode( GL_LIGHT0, osg::StateAttribute::ON );
180 
181  // lightsource
182  osg::ref_ptr<osg::LightSource> ls0 = _createLightSource();
183 
184  // translation of lightsource (with light0)
185  osg::Matrix matrix;
186  osg::ref_ptr<osg::MatrixTransform> lightTranslateNode =
187  new osg::MatrixTransform();
188 
189  matrix.makeTranslate( 0.f, -5.f, 0.f );
190  lightTranslateNode->setMatrix( matrix );
191  lightTranslateNode->addChild( ls0.get( ));
192  lightTranslateNode->setDataVariance( osg::Object::STATIC );
193 
194  root->addChild( lightTranslateNode.get( ));
195 
196  return root;
197 }

199 osg::ref_ptr< osg::LightSource > Node::_createLightSource()
200 {
201  osg::ref_ptr<osg::Light> light0 = new osg::Light();
202  light0->setLightNum( 0 ); //light0 is now GL-Light0
203  light0->setPosition( osg::Vec4( 0.f, -50.f, 0.f, 1.f ));
204  light0->setAmbient( osg::Vec4( 1.0f, 1.0f, 1.0f, 0.0f ));
205  light0->setDiffuse( osg::Vec4( 1.0f, 1.0f, 1.0f, 0.0f ));
206  light0->setDirection( osg::Vec3( 0.f, 1.f, 0.f ));
207  light0->setSpotCutoff( 30.f );
208  light0->setSpotExponent( 50.0f );
209 
210  osg::ref_ptr< osg::LightSource > ls0 = new osg::LightSource();
211  ls0->setLight( light0.get( ));
212  ls0->setDataVariance( osg::Object::STATIC );
213 
214  return ls0;
215 }
