54 bool Pipe::configInit( const eq::uint128_t& initID )
55 {
56  if( !eq::Pipe::configInit( initID ))
57  return false;
58 
59  Config* config = static_cast<Config*>( getConfig( ));
60  const InitData& initData = config->getInitData();
61  const eq::uint128_t& frameDataID = initData.getFrameDataID();
62  const bool mapped = config->mapObject( &_frameData, frameDataID );
63  LBASSERT( mapped );
64 
65  return mapped;
66 }
