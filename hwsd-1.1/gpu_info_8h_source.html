126 inline std::ostream& operator << ( std::ostream& os, const GPUInfo& info )
127 {
128  os << "GPUInfo\n" << static_cast< const NodeInfo& >( info );
129  if( !info.getName().empty( ))
130  os << " Type " << info.getName() << std::endl;
131  if( info.port != GPUInfo::defaultValue )
132  os << " Port " << info.port << std::endl;
133  if( info.device != GPUInfo::defaultValue )
134  os << " Device " << info.device << std::endl;
135  if( info.pvp[2] >0 && info.pvp[3] > 0 )
136  os << " Viewport [" << info.pvp[0] << ' ' << info.pvp[1] << ' '
137  << info.pvp[2] << ' ' << info.pvp[3] << ']' << std::endl;
138  if( info.flags != 0 )
139  os << " Flags "
140  << (info.flags&GPUInfo::FLAG_VIRTUALGL ? "VirtualGL" : "")
141  << (info.flags&GPUInfo::FLAG_VIRTUALGL_DISPLAY ? "Display" : "")
142  << std::endl;
143  return os;
144 }
