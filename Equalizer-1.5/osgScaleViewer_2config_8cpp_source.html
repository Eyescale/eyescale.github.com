65 bool Config::init()
66 {
67  // init tracker
68  std::string trackerPort = _initData.getTrackerPort();
69  if( !trackerPort.empty( ))
70  {
71  if( !_tracker.init( trackerPort ))
72  LBWARN << "Failed to initialize tracker" << std::endl;
73  else
74  {
75  // Set up position of tracking system wrt world space
76  // Note: this depends on the physical installation.
77  eq::Matrix4f matrix( eq::Matrix4f::IDENTITY );
78  matrix.scale( 1.f, 1.f, -1.f );
79  _tracker.setWorldToEmitter( matrix );
80 
81  matrix = eq::Matrix4f::IDENTITY;
82  matrix.rotate_z( -M_PI_2 );
83  _tracker.setSensorToObject( matrix );
84  LBINFO << "Tracker initialized" << std::endl;
85  }
86  }
87 
88  registerObject( &_frameData );
89  _initData.setFrameDataID( _frameData.getID( ));
90  registerObject( &_initData );
91 
92  return eq::Config::init( _initData.getID( ));
93 }

105 void Config::updateFrameData( float elapsed )
106 {
107  // Update the viewing direction based on the mouse movement
108  _cameraAngleHor += mouseViewSpeed * _pointerXDiff;
109  if( _cameraAngleHor > 2 * osg::PI || _cameraAngleHor < -2 * osg::PI )
110  _cameraAngleHor = 0.0f;
111  _cameraAngleVer += mouseViewSpeed * _pointerYDiff;
112  if( _cameraAngleVer > maxVerticalAngle )
113  _cameraAngleVer = maxVerticalAngle;
114  if( _cameraAngleVer < minVerticalAngle )
115  _cameraAngleVer = minVerticalAngle;
116  _pointerXDiff = _pointerYDiff = 0.0f;
117 
118  eq::Vector3f cameraViewingDirection;
119  cameraViewingDirection.x() = sin( _cameraAngleHor ) * sin( _cameraAngleVer );
120  cameraViewingDirection.z() = -cos( _cameraAngleHor ) * sin( _cameraAngleVer );
121  cameraViewingDirection.y() = cos( _cameraAngleVer );
122  _cameraWalkingVector = cameraViewingDirection;
123  _cameraWalkingVector.y() = 0.0f;
124 
125  // save camera data to frame data and update the camera position
126  _frameData.setCameraPosition( _frameData.getCameraPosition() +
127  _moveDirection * elapsed );
128  _frameData.setCameraLookAtPoint( _frameData.getCameraPosition() +
129  cameraViewingDirection );
130  _frameData.setCameraUpVector( eq::Vector3f( 0., 1., 0. ));
131 }

133 uint32_t Config::startFrame()
134 {
135  // update head position
136  if( _tracker.isRunning( ))
137  {
138  _tracker.update();
139  const eq::Matrix4f& headMatrix = _tracker.getMatrix();
140  _setHeadMatrix( headMatrix );

175 bool Config::handleEvent( const eq::ConfigEvent* event )
176 {
177  const float moveSpeed = .1f;
178 
179  switch( event->data.type )
180  {
181  // set mMoveDirection to a null vector after a key is released
182  // so that the updating of the camera position stops
183  case eq::Event::KEY_RELEASE:
184  if ( event->data.keyPress.key >= 261 &&
185  event->data.keyPress.key <= 266 )
186  _moveDirection = eq::Vector3f( 0, 0, 0 );
187  break;
188 
189  // change mMoveDirection when the appropriate key is pressed
190  case eq::Event::KEY_PRESS:
191  switch ( event->data.keyPress.key )
192  {
193  case eq::KC_LEFT:
194  _moveDirection = vmml::normalize( orthographicVector(
195  _cameraWalkingVector )) * moveSpeed;
196  return true;
197 
198  case eq::KC_UP:
199  _moveDirection.y() = moveSpeed;
200  return true;
201 
202  case eq::KC_RIGHT:
203  _moveDirection = -vmml::normalize( orthographicVector(
204  _cameraWalkingVector )) * moveSpeed;
205  return true;
206 
207  case eq::KC_DOWN:
208  _moveDirection.y() = -moveSpeed;
209  return true;
210 
211  case eq::KC_PAGE_UP:
212  _moveDirection = vmml::normalize( _cameraWalkingVector )
213  * moveSpeed;
214  return true;
215 
216  case eq::KC_PAGE_DOWN:
217  _moveDirection = -vmml::normalize( _cameraWalkingVector ) *
218  moveSpeed;
219  return true;
220 
221  case 's':
222  _frameData.toggleStatistics();
223  }
224  break;
225 
226  // turn left and right, up and down with mouse pointer
227  case eq::Event::CHANNEL_POINTER_MOTION:
228  if ( event->data.pointerMotion.buttons == eq::PTR_BUTTON1 &&
229  event->data.pointerMotion.x <= event->data.context.pvp.w &&
230  event->data.pointerMotion.x >= 0 &&
231  event->data.pointerMotion.y <= event->data.context.pvp.h &&
232  event->data.pointerMotion.y >= 0 )
233  {
234  _pointerXDiff += event->data.pointerMotion.dx;
235  _pointerYDiff += event->data.pointerMotion.dy;
236  return true;
237  }
238  break;
239  }
240 
241  // let Equalizer handle any events we don't handle ourselves here, like the
242  // escape key for closing the application.
243  return eq::Config::handleEvent( event );
244 }
