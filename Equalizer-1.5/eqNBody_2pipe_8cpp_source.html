47 bool Pipe::configInit( const eq::uint128_t& initID )
48 {
49  if( !eq::Pipe::configInit( initID )) {
50  return false;
51  }
52 
53  Config* config = static_cast<Config*>( getConfig() );
54  const InitData& initData = config->getInitData();
55  const eq::uint128_t frameDataID = initData.getFrameDataID();
56 
57  _data = new SharedData(config);
58  LBASSERT( _data );
59 
60  FrameData& fd = const_cast<FrameData&>(_data->getFrameData());
61  fd.init(initData.getNumBodies());
62 
63  const bool mapped = config->mapObject( &fd, frameDataID );
64  LBASSERT( mapped );
65 
66  return mapped;
67 }
