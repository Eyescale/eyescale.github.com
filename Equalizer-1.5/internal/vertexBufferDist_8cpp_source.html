178 void VertexBufferDist::applyInstanceData( co::DataIStream& is )
179 {
180  LBASSERT( !_node );
181 
182  mesh::VertexBufferNode* node = 0;
183  mesh::VertexBufferBase* base = 0;
184 
185  lunchbox::UUID leftID, rightID;
186  is >> _isRoot >> leftID >> rightID;
187 
188  if( leftID != 0 && rightID != 0 )
189  {
190  if( _isRoot )
191  {
192  mesh::VertexBufferRoot* root = new mesh::VertexBufferRoot;
193  mesh::VertexBufferData& data = root->_data;
194 
195  is >> data.vertices >> data.colors >> data.normals >> data.indices
196  >> root->_name;
197 
198  node = root;
199  _root = root;
200  }
201  else
202  {
203  LBASSERT( _root );
204  node = new mesh::VertexBufferNode;
205  }
206 
207  base = node;
208  _left = new VertexBufferDist( _root, 0 );
209  _right = new VertexBufferDist( _root, 0 );
210  co::LocalNodePtr to = getLocalNode();
211  co::NodePtr from = is.getMaster();
212  const uint32_t sync1 = to->mapObjectNB( _left, leftID,
213  co::VERSION_OLDEST, from );
214  const uint32_t sync2 = to->mapObjectNB( _right, rightID,
215  co::VERSION_OLDEST, from );
216 
217  LBCHECK( to->mapObjectSync( sync1 ));
218  LBCHECK( to->mapObjectSync( sync2 ));
219 
220  node->_left = const_cast< mesh::VertexBufferBase* >( _left->_node );
221  node->_right = const_cast< mesh::VertexBufferBase* >( _right->_node );
222  }
223  else
224  {
225  LBASSERT( !_isRoot );
226  mesh::VertexBufferData& data =
227  const_cast< mesh::VertexBufferData& >( _root->_data );
228  mesh::VertexBufferLeaf* leaf = new mesh::VertexBufferLeaf( data );
229 
230  uint64_t i1, i2, i3;
231  is >> leaf->_boundingBox[0] >> leaf->_boundingBox[1]
232  >> i1 >> i2 >> i3 >> leaf->_vertexLength;
233  leaf->_vertexStart = size_t( i1 );
234  leaf->_indexStart = size_t( i2 );
235  leaf->_indexLength = size_t( i3 );
236 
237  base = leaf;
238  }
239 
240  LBASSERT( base );
241  is >> base->_boundingSphere >> base->_range;
242 
243  _node = base;
244 }
