81 static bool _isPlyfile( const std::string& filename )
82 {
83  const size_t size = filename.length();
84  if( size < 5 )
85  return false;
86 
87  if( filename[size-4] != '.' || filename[size-3] != 'p' ||
88  filename[size-2] != 'l' || filename[size-1] != 'y' )
89  {
90  return false;
91  }
92  return true;
93 }

131 void Application::_loadModel( const eq::Strings& models )
132 {
133  eq::Strings files = models;
134  while( !files.empty( ))
135  {
136  const std::string filename = files.back();
137  files.pop_back();
138 
139  if( _isPlyfile( filename ))
140  {
141  _model = new Model;
142  if( _model->readFromFile( filename.c_str( )))
143  {
144  _modelDist = new ModelDist( _model );
145  _modelDist->registerTree( this );
146  _frameData.setModelID( _modelDist->getID( ));
147  return;
148  }
149  delete _model;
150  _model = 0;
151  }
152  else
153  {
154  const std::string basename = lunchbox::getFilename( filename );
155  if( basename == "." || basename == ".." )
156  continue;
157 
158  // recursively search directories
159  const eq::Strings subFiles = lunchbox::searchDirectory( filename,
160  ".*" );
161  for(eq::StringsCIter i = subFiles.begin(); i != subFiles.end(); ++i)
162  files.push_back( filename + '/' + *i );
163  }
164  }
165 }

180 const Model* Application::getModel( const eq::uint128_t& modelID )
181 {
182  if( modelID == 0 )
183  return 0;
184  if( _model )
185  return _model;
186  lunchbox::memoryBarrier();
187 
188  // Accessed concurrently from render threads
189  lunchbox::ScopedMutex<> mutex( _modelLock );
190  if( _model )
191  return _model;
192 
193  LBASSERT( !_modelDist );
194  _modelDist = new ModelDist;
195  Model* model = _modelDist->loadModel( getMasterNode(), this, modelID );
196  LBASSERT( model );
197  _model = model;
198 
199  return model;
200 }
