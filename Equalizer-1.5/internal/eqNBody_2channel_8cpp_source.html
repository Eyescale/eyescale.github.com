74 void Channel::frameDraw( const eq::uint128_t& frameID )
75 {
76  const eq::Range& range = getRange();
77  SharedData& sd = static_cast<Pipe*>( getPipe() )->getSharedData();
78 
79  // 1st, register the local memory
80  if( _registerMem )
81  {
82  sd.registerMemory( getRange() );
83  _registerMem = false;
84 
85  // Make sure all proxies are mapped before cont'ing
86  return;
87  }
88 
89  // 2nd, map remote memory
90  if( _mapMem )
91  {
92  sd.mapMemory();
93  _mapMem = false;
94  }
95 
96  // 3rd, synchronize the shared memory
97  sd.syncMemory();
98 
99  // 4th, update the GPU memory and run one simulation step
100  const uint32_t nBytes = sd.getNumBytes();
101  _controller->setArray( BODYSYSTEM_POSITION, sd.getPos(), nBytes );
102  _controller->setArray( BODYSYSTEM_VELOCITY, sd.getVel(), nBytes );
103  _controller->compute( sd.getTimeStep(), range );
104 
105  // 5th, draw the stars
106  eq::Channel::frameDraw( frameID );
107  _controller->draw( sd.getPos(), sd.getCol() );
108 
109 #ifndef NDEBUG
110  outlineViewport();
111 #endif
112 
113  // Finally, redistribute the newly computed data from the GPU to all
114  // interested mappers
115  sd.updateMemory( range, _controller );
116 }
