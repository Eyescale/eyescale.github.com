55 bool Controller::init( const InitData& initData, float* hPos, bool usePBO )
56 {
57  _numBodies = initData.getNumBodies();
58  _p = initData.getP();
59  _q = initData.getQ();
60  _damping = initData.getDamping();
61  _usePBO = usePBO;
62  _pointSize = 1.0f;
63 
64  // Setup p and q properly
65  if( _q * _p > 256 )
66  _p = 256 / _q;
67 
68  if ( _q == 1 && _numBodies < _p )
69  _p = _numBodies;
70 
71  if( usePBO )
72  {
73  // create the position pixel buffer objects for rendering
74  // we will actually compute directly from this memory in CUDA too
75  glGenBuffers(2, (GLuint*)_pbo);
76  for (int i = 0; i < 2; ++i)
77  {
78  glBindBuffer( GL_ARRAY_BUFFER, _pbo[i] );
79  glBufferData( GL_ARRAY_BUFFER, 4 * sizeof(float) * _numBodies,
80  hPos, GL_DYNAMIC_DRAW);
81 
82  int size = 0;
83  glGetBufferParameteriv(GL_ARRAY_BUFFER, GL_BUFFER_SIZE, (GLint*)&size);
84  if ((unsigned)size != 4 * (sizeof(float) * _numBodies)) {
85  LBERROR << "WARNING: Pixel Buffer Object allocation failed"
86  << std::endl;
87  }
88 
89  glBindBuffer(GL_ARRAY_BUFFER, 0);
90  registerGLBufferObject(_pbo[i]);
91  }
92  }
93  else
94  {
95  allocateNBodyArrays(_dPos, _numBodies * sizeof( float) * 4);
96  }
97 
98  allocateNBodyArrays(_dVel, _numBodies * sizeof( float) * 4);
99  setSoftening(0.00125f);
100  _renderer.init();
101 
102  return true;
103 }

123 {

126

128  _dPos[_currentRead], _dVel[_currentRead],
129  _pbo[_currentWrite], _pbo[_currentRead],

131  _p, _q, (_usePBO ? 1 : 0));

160 void Controller::getArray(BodyArray array, SharedDataProxy& proxy)
161 {
162  float* ddata = 0;
163  float* hdata = 0;
164  unsigned int pbo = 0;
165 
166  const unsigned int offset = proxy.getOffset();
167 
168  switch (array)
169  {
170  default:
171  case BODYSYSTEM_POSITION:
172  hdata = proxy.getPosition();
173  ddata = _dPos[ _currentRead ];
174  if (_usePBO) {
175  pbo = _pbo[ _currentRead ];
176  }
177  break;
178 
179  case BODYSYSTEM_VELOCITY:
180  hdata = proxy.getVelocity();
181  ddata = _dVel[ _currentRead ];
182  break;
183  }
184 
185  copyArrayFromDevice( hdata + offset, ddata + offset, pbo,
186  proxy.getNumBytes());
187  proxy.markDirty();
188 }

190 void Controller::setArray( BodyArray array, const float* pos,
191  unsigned int numBytes )
192 {
193  switch (array)
194  {
195  default:
196  case BODYSYSTEM_POSITION:
197  {
198  if ( !_usePBO )
199  {
200  copyArrayToDevice( _dPos[ _currentRead ], pos, numBytes );
201  return;
202  }
203 
204  unregisterGLBufferObject( _pbo[ _currentRead ] );
205  glBindBuffer(GL_ARRAY_BUFFER, _pbo[_currentRead]);
206  glBufferData(GL_ARRAY_BUFFER, numBytes, pos, GL_DYNAMIC_DRAW);
207 
208  int size = 0;
209  glGetBufferParameteriv( GL_ARRAY_BUFFER, GL_BUFFER_SIZE,
210  ( GLint* )&size );
211  if ((unsigned)size != numBytes)
212  {
213  LBERROR << " WARNING : Pixel Buffer Object download failed."
214  << " Size " << size << " does not match numBytes "
215  << numBytes << std::endl;
216  }
217  glBindBuffer(GL_ARRAY_BUFFER, 0);
218  registerGLBufferObject( _pbo[ _currentRead ] );
219  break;
220  }
221 
222  case BODYSYSTEM_VELOCITY:
223  copyArrayToDevice( _dVel[ _currentRead ], pos, numBytes );
224  break;
225  }
226 }
