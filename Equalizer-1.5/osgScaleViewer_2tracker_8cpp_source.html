51 bool Tracker::init( const std::string& port )
52 {
53 #ifdef WIN32
54  return false;
55 #else
56  if( _running )
57  {
58  LBERROR << "Duplicate tracker initialisation" << std::endl;
59  return false;
60  }
61 
62  _fd = open( port.c_str(), O_RDWR | O_EXCL );
63  if( _fd < 0 )
64  {
65  LBERROR << "Failed to open " << port << ": " << lunchbox::sysError
66  << std::endl;
67  return false;
68  }
69 
70  // configure serial port
71  struct termios termio;
72  if( tcgetattr( _fd, &termio ) != 0 )
73  {
74  LBERROR << "tcgetattr failed: " << lunchbox::sysError << std::endl;
75  close( _fd );
76  return false;
77  }
78 
79  termio.c_cflag &= ~(CSIZE|PARENB|CSTOPB|PARODD|HUPCL|CRTSCTS);
80  termio.c_cflag |= CS8|CREAD|CLOCAL;
81  termio.c_iflag &= ~(IXON|IXANY|IMAXBEL|BRKINT|IGNPAR|PARMRK|
82  INPCK|ISTRIP|INLCR|IGNCR|ICRNL);
83 #ifdef IUCLC
84  termio.c_iflag &= ~IUCLC;
85 #endif
86 
87  termio.c_iflag |= IXOFF|IGNBRK;
88  termio.c_oflag &= ~(OPOST|OCRNL|ONLCR|ONOCR|ONLRET|OFILL|OFDEL);
89 #ifdef OLCUC
90  termio.c_oflag &= ~OLCUC;
91 #endif
92 
93  termio.c_lflag &= ~(ISIG|ICANON|IEXTEN|ECHO|ECHOE|ECHOK|ECHONL|NOFLSH|
94  TOSTOP|ECHOPRT|ECHOCTL|ECHOKE);
95 #ifdef XCASE
96  termio.c_lflag &= ~XCASE;
97 #endif
98 
99  cfsetspeed( &termio, B115200 );
100  termio.c_cc[VMIN] = 26;
101  termio.c_cc[VTIME] = 1;
102 
103  if( tcsetattr( _fd, TCSANOW, &termio ) != 0)
104  {
105  LBERROR << "tcsetattr failed: " << lunchbox::sysError << std::endl;
106  close( _fd );
107  return false;
108  }
109 
110  // tell the tracker what kind of data to prepare
111  int k = write( _fd, COMMAND_POS_ANG, 1 ); //take data
112  if( k==-1 )
113  LBERROR << "Write error: " << lunchbox::sysError << std::endl;
114 
115  usleep( 10000 ); //give enough time for initializing
116 
117  if( _update( )) //try an update to see if it works
118  _running = true;
119 
120  return _running;
121 #endif
122 }

138 bool Tracker::_update()
139 {
140 #ifdef WIN32
141  return false;
142 #else
143  const ssize_t wrote = write( _fd, COMMAND_POINT, 1 ); // send data
144  if( wrote==-1 )
145  {
146  LBERROR << "Write error: " << lunchbox::sysError << std::endl;
147  return false;
148  }
149 
150  unsigned char buffer[12];
151  if( !_read( buffer, 12, 500000 ))
152  {
153  LBERROR << "Read error" << std::endl;
154  return false;
155  }
156 
157  const short xpos = (buffer[1]<<8 | buffer[0]);
158  const short ypos = (buffer[3]<<8 | buffer[2]);
159  const short zpos = (buffer[5]<<8 | buffer[4]);
160 
161  const short head = (buffer[7]<<8 | buffer[6]);
162  const short pitch = (buffer[9]<<8 | buffer[8]);
163  const short roll = (buffer[11]<<8 | buffer[10]);
164 
165  // 32640 is 360 degrees (2pi) -> scale is 1/5194.81734
166  const eq::Vector3f hpr( head / -5194.81734f + M_PI,
167  pitch / -5194.81734f + 2.0f * M_PI,
168  roll / -5194.81734f + 2.0f * M_PI );
169 
170  eq::Vector3f pos;
171 
172  // highest value for y and z position of the tracker sensor is 32639,
173  // after that it switches back to zero (and vice versa if descending values).
174  pos.x() = ypos;
175  if( pos.x() > 16320 ) //32640 / 2 = 16320
176  pos.x() -= 32640;
177 
178  pos.y() = zpos;
179  if( pos.y() > 16320 )
180  pos.y() -= 32640;
181 
182  pos.z() = xpos;
183 
184  pos /= 18000.f; // scale to meter
185 
186  // position and rotation are stored in transformation matrix
187  // and matrix is scaled to the application's units
188  _matrix = eq::Matrix4f::IDENTITY;
189  _matrix.rotate_x( hpr.x() );
190  _matrix.rotate_y( hpr.y() );
191  _matrix.rotate_z( hpr.z() );
192  _matrix.set_translation( pos );
193 
194  LBINFO << "Tracker pos " << pos << " hpr " << hpr << " = " << _matrix;
195 
196  // M = M_world_emitter * M_emitter_sensor * M_sensor_object
197  _matrix = _worldToEmitter * _matrix * _sensorToObject;
198 
199  LBINFO << "Tracker matrix " << _matrix;
200 
201  return true;
202 #endif
203 }

205 bool Tracker::_read( unsigned char* buffer, const size_t size,
206  const unsigned long int timeout )
207 {
208 #ifdef WIN32
209  return false;
210 #else
211  size_t remaining = size;
212  struct timeval tv;
213 
214  tv.tv_sec = timeout / 1000000;
215  tv.tv_usec = timeout % 1000000;
216 
217  while( remaining > 0 )
218  {
219  // wait for data
220  fd_set readfds;
221  FD_ZERO( &readfds );
222  FD_SET( _fd, &readfds );
223 
224  const int errCode = select( _fd+1, &readfds, 0, 0, &tv );
225  if( errCode == 0 )
226  {
227  LBERROR << "Error: no data from tracker" << std::endl;
228  return false;
229  }
230  if( errCode == -1 )
231  {
232  LBERROR << "Select error: " << lunchbox::sysError << std::endl;
233  return false;
234  }
235 
236  // try to read remaining bytes, returns # of read bytes
237  const ssize_t received = read( _fd, &buffer[size-remaining], remaining );
238  if( received == -1 )
239  {
240  LBERROR << "Read error: " << lunchbox::sysError << std::endl;
241  return false;
242  }
243 
244  LBASSERT( remaining >= (size_t)received );
245  remaining -= received;
246  }
247  return true;
248 #endif
249 }
