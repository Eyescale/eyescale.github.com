37 bool Window::configInitGL( const eq::uint128_t& initID )
38 {
39  if( !eq::Window::configInitGL( initID ))
40  return false;
41 
42  Window* sharedWindow = static_cast< Window* >( getSharedContextWindow( ));
43  if( sharedWindow == this || sharedWindow == 0 ) // init GL stuff
44  {
45  Node* node = static_cast< Node* >( getNode( ));
46 
47  // The code below is not thread-safe, since various STL containers are
48  // used within the OSG classes. Since this is init-only, a simple global
49  // lock is acceptable.
50  static lunchbox::Lock lock;
51  lunchbox::ScopedMutex<> mutex( lock );
52 
53  _sceneView = new SceneView;
54  _sceneView->setDefaults( SceneView::STANDARD_SETTINGS );
55  _sceneView->setFrameStamp( node->getFrameStamp( ));
56  _sceneView->init();
57  _sceneView->getState()->setContextID( node->getUniqueContextID( ));
58  _sceneView->getRenderStage()->setColorMask( new osg::ColorMask );
59 
60  osg::ref_ptr< osg::Node > model = node->getModel();
61  _sceneView->setSceneData( model );
62  }
63  else
64  _sceneView = sharedWindow->_sceneView;
65 
66  return true;
67 }
