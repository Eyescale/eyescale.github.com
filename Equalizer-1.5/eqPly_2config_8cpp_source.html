70 bool Config::init()
71 {
72  if( !_animation.isValid( ))
73  _animation.loadAnimation( _initData.getPathFilename( ));
74 
75  // init distributed objects
76  if( !_initData.useColor( ))
77  _frameData.setColorMode( COLOR_WHITE );
78 
79  _frameData.setRenderMode( _initData.getRenderMode( ));
80  registerObject( &_frameData );
81  _frameData.setAutoObsolete( getLatency( ));
82 
83  _initData.setFrameDataID( _frameData.getID( ));
84  registerObject( &_initData );
85 
86  // init config
87  if( !eq::Config::init( _initData.getID( )))
88  {
89  _deregisterData();
90  return false;
91  }
92 
93  _loadModels();
94  _registerModels();
95 
96  // init tracker
97  if( !_initData.getTrackerPort().empty( ))
98  {
99  if( !_tracker.init( _initData.getTrackerPort() ))
100  LBWARN << "Failed to initialize tracker" << std::endl;
101  else
102  {
103  // Set up position of tracking system wrt world space
104  // Note: this depends on the physical installation.
105  eq::Matrix4f m( eq::Matrix4f::IDENTITY );
106  m.scale( 1.f, 1.f, -1.f );
107  _tracker.setWorldToEmitter( m );
108 
109  m = eq::Matrix4f::IDENTITY;
110  m.rotate_z( -M_PI_2 );
111  _tracker.setSensorToObject( m );
112  LBINFO << "Tracker initialized" << std::endl;
113  }
114  }
115 
116  const eq::Canvases& canvases = getCanvases();
117  if( canvases.empty( ))
118  _currentCanvas = 0;
119  else
120  _currentCanvas = canvases.front();
121 
122  _setMessage( "Welcome to eqPly\nPress F1 for help" );
123  return true;
124 }

138 static bool _isPlyfile( const std::string& filename )
139 {
140  const size_t size = filename.length();
141  if( size < 5 )
142  return false;
143 
144  if( filename[size-4] != '.' || filename[size-3] != 'p' ||
145  filename[size-2] != 'l' || filename[size-1] != 'y' )
146  {
147  return false;
148  }
149  return true;
150 }

153 void Config::_loadModels()
154 {
155  if( !_models.empty( )) // only load on the first config run
156  return;
157 
158  eq::Strings filenames = _initData.getFilenames();
159  while( !filenames.empty( ))
160  {
161  const std::string filename = filenames.back();
162  filenames.pop_back();
163 
164  if( _isPlyfile( filename ))
165  {
166  Model* model = new Model;
167 
168  if( _initData.useInvertedFaces() )
169  model->useInvertedFaces();
170 
171  if( !model->readFromFile( filename.c_str( )))
172  {
173  LBWARN << "Can't load model: " << filename << std::endl;
174  delete model;
175  }
176  else
177  _models.push_back( model );
178  }
179  else
180  {
181  const std::string basename = lunchbox::getFilename( filename );
182  if( basename == "." || basename == ".." )
183  continue;
184 
185  // recursively search directories
186  const eq::Strings subFiles = lunchbox::searchDirectory( filename,
187  ".*" );
188  for(eq::StringsCIter i = subFiles.begin(); i != subFiles.end(); ++i)
189  filenames.push_back( filename + '/' + *i );
190  }
191  }
192 }

194 void Config::_registerModels()
195 {
196  // Register distribution helpers on each config run
197  const bool createDist = _modelDist.empty(); //first run, create distributors
198  const size_t nModels = _models.size();
199  LBASSERT( createDist || _modelDist.size() == nModels );
200 
201  for( size_t i = 0; i < nModels; ++i )
202  {
203  const Model* model = _models[i];
204  ModelDist* modelDist = 0;
205  if( createDist )
206  {
207  modelDist = new ModelDist( model );
208  _modelDist.push_back( modelDist );
209  }
210  else
211  modelDist = _modelDist[i];
212 
213  modelDist->registerTree( getClient( ));
214  LBASSERT( modelDist->isAttached() );
215 
216  _frameData.setModelID( modelDist->getID( ));
217  }
218 
219  LBASSERT( _modelDist.size() == nModels );
220 
221  if( !_modelDist.empty( ))
222  {
223  ModelAssigner assigner( _modelDist );
224  accept( assigner );
225  }
226 }

265 const Model* Config::getModel( const eq::uint128_t& modelID )
266 {
267  if( modelID == 0 )
268  return 0;
269 
270  // Protect if accessed concurrently from multiple pipe threads
271  const eq::Node* node = getNodes().front();
272  const bool needModelLock = (node->getPipes().size() > 1);
273  lunchbox::ScopedWrite _mutex( needModelLock ? &_modelLock : 0 );
274 
275  const size_t nModels = _models.size();
276  LBASSERT( _modelDist.size() == nModels );
277 
278  for( size_t i = 0; i < nModels; ++i )
279  {
280  const ModelDist* dist = _modelDist[ i ];
281  if( dist->getID() == modelID )
282  return _models[ i ];
283  }
284 
285  _modelDist.push_back( new ModelDist );
286  Model* model = _modelDist.back()->loadModel( getApplicationNode(),
287  getClient(), modelID );
288  LBASSERT( model );
289  _models.push_back( model );
290 
291  return model;
292 }

303 void Config::_updateData()
304 {
305  // update head position
306  if( _tracker.isRunning() )
307  {
308  _tracker.update();
309  const eq::Matrix4f& headMatrix = _tracker.getMatrix();
310  _setHeadMatrix( headMatrix );
311  }
312 
313  // update camera
314  if( _animation.isValid( ))
315  {
316  const eq::Vector3f& modelRotation = _animation.getModelRotation();
317  const CameraAnimation::Step& curStep = _animation.getNextStep();
318 
319  _frameData.setModelRotation( modelRotation);
320  _frameData.setRotation( curStep.rotation );
321  _frameData.setCameraPosition( curStep.position );
322  }
323  else
324  {
325  if( _frameData.usePilotMode())
326  _frameData.spinCamera( -0.001f * _spinX, -0.001f * _spinY );
327  else
328  _frameData.spinModel( -0.001f * _spinX, -0.001f * _spinY, 0.f );
329 
330  _frameData.moveCamera( 0.0f, 0.0f, 0.001f * _advance );
331  }
332 
333  // idle mode
334  if( isIdleAA( ))
335  {
336  LBASSERT( _numFramesAA > 0 );
337  _frameData.setIdle( true );
338  }
339  else
340  _frameData.setIdle( false );
341 
342  _numFramesAA = 0;
343 }

360 bool Config::_needNewFrame()
361 {
362  if( _messageTime > 0 )
363  {
364  if( getTime() - _messageTime > 2000 ) // reset message after two seconds
365  {
366  _messageTime = 0;
367  _frameData.setMessage( "" );
368  }
369  return true;
370  }
371 
372  return ( _spinX != 0 || _spinY != 0 || _advance != 0 ||
373  _tracker.isRunning() || _redraw );

376 bool Config::handleEvent( eq::EventICommand command )
377 {
378  switch( command.getEventType( ))
379  {
380  case eq::Event::KEY_PRESS:
381  {
382  const eq::Event& event = command.get< eq::Event >();
383  if( _handleKeyEvent( event.keyPress ))
384  {
385  _redraw = true;
386  return true;
387  }
388  break;
389  }
390 
391  case eq::Event::CHANNEL_POINTER_BUTTON_PRESS:
392  {
393  const eq::Event& event = command.get< eq::Event >();
394  const eq::uint128_t& viewID = event.context.view.identifier;
395  _frameData.setCurrentViewID( viewID );
396  if( viewID == 0 )
397  {
398  _currentCanvas = 0;
399  return false;
400  }
401 
402  const View* view = _getCurrentView();
403  const eq::Layout* layout = view->getLayout();
404  const eq::Canvases& canvases = getCanvases();
405  for( eq::CanvasesCIter i = canvases.begin();
406  i != canvases.end(); ++i )
407  {
408  eq::Canvas* canvas = *i;
409  const eq::Layout* canvasLayout = canvas->getActiveLayout();
410 
411  if( canvasLayout == layout )
412  {
413  _currentCanvas = canvas;
414  return true;
415  }
416  }
417  return true;
418  }
419 
420  case eq::Event::CHANNEL_POINTER_BUTTON_RELEASE:
421  {
422  const eq::Event& event = command.get< eq::Event >();
423  const eq::PointerEvent& releaseEvent =
424  event.pointerButtonRelease;
425  if( releaseEvent.buttons == eq::PTR_BUTTON_NONE)
426  {
427  if( releaseEvent.button == eq::PTR_BUTTON1 )
428  {
429  _spinX = releaseEvent.dy;
430  _spinY = releaseEvent.dx;
431  _redraw = true;
432  return true;
433  }
434  if( releaseEvent.button == eq::PTR_BUTTON2 )
435  {
436  _advance = -releaseEvent.dy;
437  _redraw = true;
438  return true;
439  }
440  }
441  break;
442  }
443  case eq::Event::CHANNEL_POINTER_MOTION:
444  {
445  const eq::Event& event = command.get< eq::Event >();
446  switch( event.pointerMotion.buttons )
447  {
448  case eq::PTR_BUTTON1:
449  _spinX = 0;
450  _spinY = 0;
451 
452  if( _frameData.usePilotMode())
453  _frameData.spinCamera(
454  -0.005f * event.pointerMotion.dy,
455  -0.005f * event.pointerMotion.dx );
456  else
457  _frameData.spinModel(
458  -0.005f * event.pointerMotion.dy,
459  -0.005f * event.pointerMotion.dx, 0.f );
460  _redraw = true;
461  return true;
462 
463  case eq::PTR_BUTTON2:
464  _advance = -event.pointerMotion.dy;
465  _frameData.moveCamera( 0.f, 0.f, .005f * _advance );
466  _redraw = true;
467  return true;
468 
469  case eq::PTR_BUTTON3:
470  _frameData.moveCamera( .0005f * event.pointerMotion.dx,
471  -.0005f * event.pointerMotion.dy,
472  0.f );
473  _redraw = true;
474  return true;
475  }
476  break;
477  }
478 
479  case eq::Event::CHANNEL_POINTER_WHEEL:
480  {
481  const eq::Event& event = command.get< eq::Event >();
482  _frameData.moveCamera( -0.05f * event.pointerWheel.yAxis,
483  0.f,
484  0.05f * event.pointerWheel.xAxis );
485  _redraw = true;
486  return true;
487  }
488 
489  case eq::Event::MAGELLAN_AXIS:
490  {
491  const eq::Event& event = command.get< eq::Event >();
492  _spinX = 0;
493  _spinY = 0;
494  _advance = 0;
495  _frameData.spinModel( 0.0001f * event.magellan.xRotation,
496  0.0001f * event.magellan.yRotation,
497  0.0001f * event.magellan.zRotation );
498  _frameData.moveCamera( 0.0001f * event.magellan.xAxis,
499  0.0001f * event.magellan.yAxis,
500  0.0001f * event.magellan.zAxis );
501  _redraw = true;
502  return true;
503  }
504 
505  case eq::Event::MAGELLAN_BUTTON:
506  {
507  const eq::Event& event = command.get< eq::Event >();
508  if( event.magellan.button == eq::PTR_BUTTON1 )
509  _frameData.toggleColorMode();
510 
511  _redraw = true;
512  return true;
513  }
514 
515  case eq::Event::WINDOW_EXPOSE:
516  case eq::Event::WINDOW_RESIZE:
517  case eq::Event::WINDOW_CLOSE:
518  case eq::Event::VIEW_RESIZE:
519  _redraw = true;
520  break;
521 
522  case IDLE_AA_LEFT:
523  if( _useIdleAA )
524  {
525  const int32_t steps = command.get< int32_t >();
526  _numFramesAA = LB_MAX( _numFramesAA, steps );
527  }
528  else
529  _numFramesAA = 0;
530  return false;
531 
532  default:
533  break;
534  }
535 
536  _redraw |= eq::Config::handleEvent( command );
537  return _redraw;
538 }
