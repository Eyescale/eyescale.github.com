89  PixelViewport& apply( const Viewport& vp )
90  {
91  // honor position over size to avoid rounding artifacts
92  const int32_t xEnd = x + int32_t((vp.x+vp.w)*w);
93  const int32_t yEnd = y + int32_t((vp.y+vp.h)*h);
94 
95  x += int32_t( w * vp.x );
96  y += int32_t( h * vp.y );
97  w = xEnd - x;
98  h = yEnd - y;
99  return *this;
100  }

103  void apply( const Pixel& pixel )
104  {
105  if( pixel.w > 1 )
106  {
107  int32_t newWidth = w / pixel.w;
108  // This would be the correct thing to do, but it would
109  // require frustum adaptations in CUV::_computeFrustum:
110  //if( w - ( newWidth * pixel.w ) > pixel.x )
111  if( w - ( newWidth * pixel.w ) != 0 )
112  ++newWidth;
113 
114  w = newWidth;
115  }
116  if( pixel.h > 1 )
117  {
118  int32_t newHeight = h / pixel.h;
119  // This would be the correct thing to do, but it would
120  // require frustum adaptations in CUV::_computeFrustum:
121  // if( w - ( newWidth * pixel.h ) > pixel.y )
122  if( h - ( newHeight * pixel.h ) != 0 )
123  ++newHeight;
124 
125  h = newHeight;
126  }
127  }

251  PixelViewport& merge( const PixelViewport& rhs )
252  {
253  if( *this == rhs || !rhs.isValid( ))
254  return *this;
255 
256  if( !hasArea( ))
257  {
258  *this = rhs;
259  return *this;
260  }

277  void intersect( const PixelViewport& rhs )
278  {
279  if( *this == rhs )
280  return;
281 
282  if( !rhs.isValid() || !isValid() )
283  {
284  invalidate();
285  return;
286  }
287 
288  if( !rhs.hasArea() || !hasArea() )
289  {
290  x = 0;
291  y = 0;
292  w = 0;
293  h = 0;
294  return;
295  }
296 
297  const int32_t sEx = x + w;
298  const int32_t sEy = y + h;
299  const int32_t dEx = rhs.x + rhs.w;
300  const int32_t dEy = rhs.y + rhs.h;
301 
302  x = LB_MAX( x, rhs.x );
303  y = LB_MAX( y, rhs.y );
304  w = LB_MIN( sEx, dEx ) - x;
305  h = LB_MIN( sEy, dEy ) - y;
