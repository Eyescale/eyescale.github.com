48  void ref( const void* holder = 0 ) const
49  {
50 #ifndef NDEBUG
51  LBASSERT( !_hasBeenDeleted );
52 #endif
53  ++_refCount;
54 
55 #ifdef LUNCHBOX_REFERENCED_DEBUG
56  if( holder )
57  {
58  std::stringstream cs;
59  cs << backtrace;
60  ScopedMutex<> referencedMutex( _holders );
61  HolderHash::iterator i = _holders->find( holder );
62  LBASSERTINFO( i == _holders->end(), i->second );
63  _holders.data[ holder ] = cs.str();
64  }
65 #endif
66  }

74  void unref( const void* holder = 0 ) const
75  {
76 #ifndef NDEBUG
77  LBASSERT( !_hasBeenDeleted );
78 #endif
79  LBASSERT( _refCount > 0 );
80  const bool deleteMe = (--_refCount==0);
81  if( deleteMe )
82  deleteReferenced( this );
83 #ifdef LUNCHBOX_REFERENCED_DEBUG
84  else if( holder )
85  {
86  ScopedMutex<> referencedMutex( _holders );
87  HolderHash::iterator i = _holders->find( holder );
88  LBASSERT( i != _holders->end( ));
89  _holders->erase( i );
90  }
91 #endif
92  }

99  {

102  ScopedMutex<> referencedMutex( _holders );

104  i != _holders->end(); ++i )
