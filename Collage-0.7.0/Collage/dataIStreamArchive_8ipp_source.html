00038 template< typename T >
00039 typename boost::enable_if< boost::is_integral<T> >::type
00040 DataIStreamArchive::load( T& t )
00041 {
00042 #if BOOST_VERSION < 104800
00043     namespace bs = boost::detail;
00044 #else
00045     namespace bs = boost::spirit::detail;
00046 #endif
00047 
00048     // get the number of bytes in the stream
00049     if( signed char size = _loadSignedChar( ))
00050     {
00051         // check for negative value in unsigned type
00052         if( size < 0 && boost::is_unsigned<T>::value )
00053             throw DataStreamArchiveException();
00054 
00055         // check that our type T is large enough
00056         else if( (unsigned)abs(size) > sizeof(T))
00057             throw DataStreamArchiveException( size );
00058 
00059         // reconstruct the value
00060         T temp = size < 0 ? -1 : 0;
00061         load_binary( &temp, abs(size));
00062 
00063         // load the value from little endian - is is then converted
00064         // to the target type T and fits it because size <= sizeof(T)
00065         t = bs::load_little_endian<T, sizeof(T)>( &temp );
00066     }
00067     else
00068         // zero optimization
00069         t = 0;
00070 }

00072 template< typename T >
00073 typename boost::enable_if<boost::is_floating_point<T> >::type
00074 DataIStreamArchive::load( T& t )
00075 {
00076     namespace fp = boost::spirit::math;
00077 
00078     typedef typename fp::detail::fp_traits<T>::type traits;
00079 
00080     // if you end here there are three possibilities:
00081     // 1. you're serializing a long double which is not portable
00082     // 2. you're serializing a double but have no 64 bit integer
00083     // 3. your machine is using an unknown floating point format
00084     // after reading the note above you still might decide to
00085     // deactivate this static assert and try if it works out.
00086     typename traits::bits bits;
00087     BOOST_STATIC_ASSERT( sizeof(bits) == sizeof(T));
00088     BOOST_STATIC_ASSERT( std::numeric_limits<T>::is_iec559 );
00089 
00090     load( bits );
00091     traits::set_bits( t, bits );
00092 
00093     // if the no_infnan flag is set we must throw here
00094     if( get_flags() & no_infnan && !fp::isfinite( t ))
00095         throw DataStreamArchiveException( t );
00096 
00097     // if you end here your floating point type does not support
00098     // denormalized numbers. this might be the case even though
00099     // your type conforms to IEC 559 (and thus to IEEE 754)
00100     if( std::numeric_limits<T>::has_denorm == std::denorm_absent &&
00101         fp::fpclassify(t) == (int)FP_SUBNORMAL ) // GCC4
00102     {
00103         throw DataStreamArchiveException( t );
00104     }
00105 }
