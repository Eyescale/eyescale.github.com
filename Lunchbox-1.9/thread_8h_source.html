184 // These thread-safety checks are for development purposes, to check that
185 // certain objects are properly used within the framework. Leaving them enabled
186 // during application development may cause false positives, e.g., when
187 // threadsafety is ensured outside of the objects by the application.

194 #define LB_TS_VAR( NAME ) \
195  public: \
196  struct NAME ## Struct \
197  { \
198  NAME ## Struct () \
199  : extMutex( false ) \
200  {} \
201  mutable lunchbox::ThreadID id; \
202  mutable std::string name; \
203  bool extMutex; \
204  mutable lunchbox::ThreadID inRegion; \
205  } NAME; \

208 #ifdef LB_CHECK_THREADSAFETY
209 # define LB_TS_RESET( NAME ) NAME.id = lunchbox::ThreadID();
210 
211 # define LB_TS_THREAD( NAME ) \
212  { \
213  if( NAME.id == lunchbox::ThreadID( )) \
214  { \
215  NAME.id = lunchbox::Thread::getSelfThreadID(); \
216  NAME.name = lunchbox::Log::instance().getThreadName(); \
217  LBVERB << "Functions for " << #NAME \
218  << " locked to this thread" << std::endl; \
219  } \
220  if( !NAME.extMutex && NAME.id != lunchbox::Thread::getSelfThreadID( )) \
221  { \
222  LBERROR << "Threadsafety check for " << #NAME \
223  << " failed on object of type " \
224  << lunchbox::className( this ) << ", thread " \
225  << lunchbox::Thread::getSelfThreadID() << " (" \
226  << lunchbox::Log::instance().getThreadName() << ") != " \
227  << NAME.id << " (" << NAME.name << ")" << std::endl; \
228  LBABORT( "Non-threadsave code called from two threads" ); \
229  } \
230  }
231 
232 # define LB_TS_NOT_THREAD( NAME ) \
233  { \
234  if( !NAME.extMutex && NAME.id != lunchbox::ThreadID( )) \
235  { \
236  if( NAME.id == lunchbox::Thread::getSelfThreadID( )) \
237  { \
238  LBERROR << "Threadsafety check for not " << #NAME \
239  << " failed on object of type " \
240  << lunchbox::className( this ) << std::endl; \
241  LBABORT( "Code called from wrong thread" ); \
242  } \
243  } \
244  }

247 template< typename T > class ScopedThreadCheck : public NonCopyable
248 {
249 public:
250  explicit ScopedThreadCheck( const T& data )
251  : _data( data )
252  {
253  LBASSERTINFO( data.inRegion == lunchbox::ThreadID() ||
254  data.inRegion == lunchbox::Thread::getSelfThreadID(),
255  "Another thread already in critical region" );
256  data.inRegion = lunchbox::Thread::getSelfThreadID();
257  }
258 
259  ~ScopedThreadCheck()
260  {
261  LBASSERTINFO( _data.inRegion == lunchbox::ThreadID() ||
262  _data.inRegion == lunchbox::Thread::getSelfThreadID(),
263  "Another thread entered critical region" );
264  _data.inRegion = lunchbox::ThreadID();
265  }
266 private:
267  const T& _data;
268 };
