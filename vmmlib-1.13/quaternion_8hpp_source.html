181 template <typename T>
182 template <size_t M>
183 Quaternion<T>::Quaternion(const Matrix<M, M, T>& rot,
184  typename std::enable_if<M >= 3>::type*)
185 {
186  const T trace = rot(0, 0) + rot(1, 1) + rot(2, 2) + T(1);
187  static const T TRACE_EPSILON = 1e-5;
188 
189  // very small traces may introduce a big numerical error
190  if (trace > TRACE_EPSILON)
191  {
192  T s = 0.5 / std::sqrt(trace);
193  array[0] = rot(2, 1) - rot(1, 2);
194  array[0] *= s;
195 
196  array[1] = rot(0, 2) - rot(2, 0);
197  array[1] *= s;
198 
199  array[2] = rot(1, 0) - rot(0, 1);
200  array[2] *= s;
201 
202  array[3] = 0.25 / s;
203  }
204  else
205  {
206  const vector<3, T> diag(rot(0, 0), rot(1, 1), rot(2, 2));
207  const size_t largest = diag.find_max_index();
208 
209  // 0, 0 is largest
210  if (largest == 0)
211  {
212  const T s =
213  0.5 / std::sqrt(T(1) + rot(0, 0) - rot(1, 1) - rot(2, 2));
214  array[0] = 0.25 / s;
215 
216  array[1] = rot(0, 1) + rot(1, 0);
217  array[1] *= s;
218 
219  array[2] = rot(0, 2) + rot(2, 0);
220  array[2] *= s;
221 
222  array[3] = rot(1, 2) - rot(2, 1);
223  array[3] *= s;
224  }
225  else if (largest == 1)
226  {
227  const T s =
228  0.5 / std::sqrt(T(1) + rot(1, 1) - rot(0, 0) - rot(2, 2));
229  array[0] = rot(0, 1) + rot(1, 0);
230  array[0] *= s;
231 
232  array[1] = 0.25 / s;
233 
234  array[2] = rot(1, 2) + rot(2, 1);
235  array[2] *= s;
236 
237  array[3] = rot(0, 2) - rot(2, 0);
238  array[3] *= s;
239  }
240  // 2, 2 is largest
241  else if (largest == 2)
242  {
243  const T s =
244  0.5 / std::sqrt(T(1) + rot(2, 2) - rot(0, 0) - rot(1, 1));
245  array[0] = rot(0, 2) + rot(2, 0);
246  array[0] *= s;
247 
248  array[1] = rot(1, 2) + rot(2, 1);
249  array[1] *= s;
250 
251  array[2] = 0.25 / s;
252 
253  array[3] = rot(0, 1) - rot(1, 0);
254  array[3] *= s;
255  }
256  else
257  {
258  throw std::runtime_error("no clue why, but should not get here");
259  }
260  }
261 }

263 template <typename T>
264 Quaternion<T>::Quaternion(const T angle, vector<3, T> axis)
265 {
266  axis.normalize();
267  const T sinAngle2 = std::sin(angle * 0.5);
268  array[0] = axis.x() * sinAngle2;
269  array[1] = axis.y() * sinAngle2;
270  array[2] = axis.z() * sinAngle2;
271  array[3] = std::cos(angle * 0.5);
272 }

360 // Grassmann product
361 template <typename T>
362 void Quaternion<T>::operator*=(const Quaternion<T>& q)
363 {
364  // optimized version, 7 less mul, but 15 more add/subs
365  // after Henrik Engstrom, from a gamedev.net article.
366  const T& a_ = array[3];
367  const T& b_ = array[0];
368  const T& c = array[1];
369  const T& d = array[2];
370  const T& _x = q.array[3];
371  const T& _y = q.array[0];
372  const T& _z = q.array[1];
373  const T& _w = q.array[2];
374 
375  const T tmp_00 = (d - c) * (_z - _w);
376  const T tmp_01 = (a_ + b_) * (_x + _y);
377  const T tmp_02 = (a_ - b_) * (_z + _w);
378  const T tmp_03 = (c + d) * (_x - _y);
379  const T tmp_04 = (d - b_) * (_y - _z);
380  const T tmp_05 = (d + b_) * (_y + _z);
381  const T tmp_06 = (a_ + c) * (_x - _w);
382  const T tmp_07 = (a_ - c) * (_x + _w);
383  const T tmp_08 = tmp_05 + tmp_06 + tmp_07;
384  const T tmp_09 = 0.5 * (tmp_04 + tmp_08);
385 
386  array[0] = tmp_01 + tmp_09 - tmp_08;
387  array[1] = tmp_02 + tmp_09 - tmp_07;
388  array[2] = tmp_03 + tmp_09 - tmp_06;
389  array[3] = tmp_00 + tmp_09 - tmp_05;
390 }

457 template <typename T>
458 Matrix<3, 3, T> Quaternion<T>::getRotationMatrix() const
459 {
460  const T w2 = array[3] * array[3];
461  const T x2 = array[0] * array[0];
462  const T y2 = array[1] * array[1];
463  const T z2 = array[2] * array[2];
464  const T wx = array[3] * array[0];
465  const T wy = array[3] * array[1];
466  const T wz = array[3] * array[2];
467  const T xy = array[0] * array[1];
468  const T xz = array[0] * array[2];
469  const T yz = array[1] * array[2];
470 
471  Matrix<3, 3, T> matrix;
472  matrix(0, 0) = w2 + x2 - y2 - z2;
473  matrix(0, 1) = 2. * (xy - wz);
474  matrix(0, 2) = 2. * (xz + wy);
475  matrix(1, 0) = 2. * (xy + wz);
476  matrix(1, 1) = w2 - x2 + y2 - z2;
477  matrix(1, 2) = 2. * (yz - wx);
478  matrix(2, 0) = 2. * (xz - wy);
479  matrix(2, 1) = 2. * (yz + wx);
480  matrix(2, 2) = w2 - x2 - y2 + z2;
481  return matrix;
482 }
