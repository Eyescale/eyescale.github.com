79 void ThreadPool::work()
80 {
81  for( ;; )
82  {
83  std::function< void() > task;
84  {
85  std::unique_lock< std::mutex > lock( _tasksMutex );
86  _waitCondition.wait( lock, [this] { return _stop || !_tasks.empty(); });
87  if( _stop )
88  return;
89  task = std::move( _tasks.front());
90  _tasks.pop();
91  }
92  task();
93  }
94 }
95 }
