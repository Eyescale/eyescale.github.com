38  if( !eq::Node::configInit( initID ))
39  return false;
40 
41  // All render data is static or multi-buffered, we can run asynchronously
42  if( getIAttribute( IATTR_THREAD_MODEL ) == eq::UNDEFINED )
43  setIAttribute( IATTR_THREAD_MODEL, eq::ASYNC );
44 
45  Config* config = static_cast< Config* >( getConfig( ));
46  if( !isApplicationNode() && !config->loadInitData( initID ))
47  {
48  sendError( ERROR_EVOLVE_MAPOBJECT_FAILED );
49  return false;
50  }
51  return true;
52 }
