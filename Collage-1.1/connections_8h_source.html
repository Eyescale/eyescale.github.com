47 {
48  Connections result;
49  typedef stde::hash_map< ConstConnectionDescriptionPtr, NodePtr,
50  lunchbox::hashRefPtr< const ConnectionDescription > >
51  MCNodes;
52  MCNodes mcNodes; // first node using a multicast connection
53 
54  typedef std::set< ConstConnectionDescriptionPtr > MCSet;
55  MCSet mcSet; // multicast connection is added
56 
57  for( Nodes::const_iterator i = nodes.begin(); i != nodes.end(); ++i )
58  {
59  NodePtr node = *i;
60  ConnectionPtr connection = node->getConnection( multicast );
61  LBASSERT( connection );
62  if( !connection )
63  continue;
64 
65  if( connection->isMulticast( ))
66  {
67  ConstConnectionDescriptionPtr desc = connection->getDescription();
68  if( mcSet.find( desc ) != mcSet.end( )) // already added
69  continue;
70 
71  MCNodes::iterator j = mcNodes.find( desc );
72  if( j == mcNodes.end( ))
73  {
74  // first appearance of multicast connection
75  mcNodes[ desc ] = node;
76  continue;
77  }
78 
79  mcSet.insert( desc ); // mark as added
80  mcNodes.erase( j );
81  }
82 
83  result.push_back( connection );
84  }
85 
86  // Add unicast connections for multicast node connections seen only once
87  for( MCNodes::iterator i = mcNodes.begin(); i != mcNodes.end(); ++i )
88  {
89  ConnectionPtr connection = i->second->getConnection();
90  LBASSERT( connection.isValid( ));
91 
92  if( connection.isValid( ))
93  result.push_back( connection );
94  }
95  return result;
96 }
