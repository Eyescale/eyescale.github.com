186  LBASSERTINFO( nElems < LB_BIT48,
187  "Out-of-sync co::DataIStream: " << nElems << " elements?" );
188  value.resize( size_t( nElems ));

199  template< class T > void _swap( Array< T > array ) const
200  {
201  if( !isSwapping( ))
202  return;
203 #pragma omp parallel for
204  for( ssize_t i = 0; i < ssize_t( array.num ); ++i )
205  swap( array.data[i] );
