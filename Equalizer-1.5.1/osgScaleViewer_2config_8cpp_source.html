61 bool Config::init()
62 {
63  // init tracker
64  std::string trackerPort = _initData.getTrackerPort();
65  if( !trackerPort.empty( ))
66  {
67  if( !_tracker.init( trackerPort ))
68  LBWARN << "Failed to initialize tracker" << std::endl;
69  else
70  {
71  // Set up position of tracking system wrt world space
72  // Note: this depends on the physical installation.
73  eq::Matrix4f matrix( eq::Matrix4f::IDENTITY );
74  matrix.scale( 1.f, 1.f, -1.f );
75  _tracker.setWorldToEmitter( matrix );
76 
77  matrix = eq::Matrix4f::IDENTITY;
78  matrix.rotate_z( -M_PI_2 );
79  _tracker.setSensorToObject( matrix );
80  LBINFO << "Tracker initialized" << std::endl;
81  }
82  }
83 
84  registerObject( &_frameData );
85  _initData.setFrameDataID( _frameData.getID( ));
86  registerObject( &_initData );
87 
88  return eq::Config::init( _initData.getID( ));
89 }

101 void Config::updateFrameData( float elapsed )
102 {
103  // Update the viewing direction based on the mouse movement
104  _cameraAngleHor += mouseViewSpeed * _pointerXDiff;
105  if( _cameraAngleHor > 2 * osg::PI || _cameraAngleHor < -2 * osg::PI )
106  _cameraAngleHor = 0.0f;
107  _cameraAngleVer += mouseViewSpeed * _pointerYDiff;
108  if( _cameraAngleVer > maxVerticalAngle )
109  _cameraAngleVer = maxVerticalAngle;
110  if( _cameraAngleVer < minVerticalAngle )
111  _cameraAngleVer = minVerticalAngle;
112  _pointerXDiff = _pointerYDiff = 0.0f;
113 
114  eq::Vector3f cameraViewingDirection;
115  cameraViewingDirection.x() = sin( _cameraAngleHor ) * sin( _cameraAngleVer );
116  cameraViewingDirection.z() = -cos( _cameraAngleHor ) * sin( _cameraAngleVer );
117  cameraViewingDirection.y() = cos( _cameraAngleVer );
118  _cameraWalkingVector = cameraViewingDirection;
119  _cameraWalkingVector.y() = 0.0f;
120 
121  // save camera data to frame data and update the camera position
122  _frameData.setCameraPosition( _frameData.getCameraPosition() +
123  _moveDirection * elapsed );
124  _frameData.setCameraLookAtPoint( _frameData.getCameraPosition() +
125  cameraViewingDirection );
126  _frameData.setCameraUpVector( eq::Vector3f( 0., 1., 0. ));
127 }

130 {

171 bool Config::handleEvent( eq::EventICommand command )
172 {
173  const float moveSpeed = .1f;
174 
175  const eq::Event& event = command.get< eq::Event >();
176  switch( command.getEventType( ))
177  {
178  // set mMoveDirection to a null vector after a key is released
179  // so that the updating of the camera position stops
180  case eq::Event::KEY_RELEASE:
181  if ( event.keyPress.key >= 261 &&
182  event.keyPress.key <= 266 )
183  _moveDirection = eq::Vector3f( 0, 0, 0 );
184  break;
185 
186  // change mMoveDirection when the appropriate key is pressed
187  case eq::Event::KEY_PRESS:
188  switch ( event.keyPress.key )
189  {
190  case eq::KC_LEFT:
191  _moveDirection = vmml::normalize( orthographicVector(
192  _cameraWalkingVector )) * moveSpeed;
193  return true;
194 
195  case eq::KC_UP:
196  _moveDirection.y() = moveSpeed;
197  return true;
198 
199  case eq::KC_RIGHT:
200  _moveDirection = -vmml::normalize( orthographicVector(
201  _cameraWalkingVector )) * moveSpeed;
202  return true;
203 
204  case eq::KC_DOWN:
205  _moveDirection.y() = -moveSpeed;
206  return true;
207 
208  case eq::KC_PAGE_UP:
209  _moveDirection = vmml::normalize( _cameraWalkingVector )
210  * moveSpeed;
211  return true;
212 
213  case eq::KC_PAGE_DOWN:
214  _moveDirection = -vmml::normalize( _cameraWalkingVector ) *
215  moveSpeed;
216  return true;
217 
218  case 's':
219  _frameData.toggleStatistics();
220  }
221  break;
222 
223  // turn left and right, up and down with mouse pointer
224  case eq::Event::CHANNEL_POINTER_MOTION:
225  if ( event.pointerMotion.buttons == eq::PTR_BUTTON1 &&
226  event.pointerMotion.x <= event.context.pvp.w &&
227  event.pointerMotion.x >= 0 &&
228  event.pointerMotion.y <= event.context.pvp.h &&
229  event.pointerMotion.y >= 0 )
230  {
231  _pointerXDiff += event.pointerMotion.dx;
232  _pointerYDiff += event.pointerMotion.dy;
233  return true;
234  }
235  break;
236  }
237 
238  // let Equalizer handle any events we don't handle ourselves here, like the
239  // escape key for closing the application.
240  return eq::Config::handleEvent( command );
241 }
