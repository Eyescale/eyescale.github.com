115  inline std::ostream& operator << ( std::ostream& os, const GPUInfo& info )
116  {
117  if( !info.getName().empty( ))

119  if( !info.hostname.empty( ))
120  os << "hostname " << info.hostname << std::endl;
121  if( !info.session.empty() && info.session != "local" )
122  os << "session " << info.session << std::endl;
123  if( info.port != GPUInfo::defaultValue )
124  os << "port " << info.port << std::endl;
125  if( info.device != GPUInfo::defaultValue )
126  os << "device " << info.device << std::endl;
127  if( info.pvp[2] >0 && info.pvp[3] > 0 )
128  os << "viewport [" << info.pvp[0] << ' ' << info.pvp[1] << ' '
129  << info.pvp[2] << ' ' << info.pvp[3] << ']' << std::endl;
130  if( info.flags != 0 )
131  os << "flags "
132  << (info.flags&GPUInfo::FLAG_VIRTUALGL ? "VirtualGL" : "")
133  << (info.flags&GPUInfo::FLAG_VIRTUALGL_DISPLAY ? "Display" : "")
134  << std::endl;
135  return os;
