240  template< class C, class PKG, class S > inline void
241  Object::commitChild( C* child, S* sender, const uint32_t incarnation )
242  {
243  if( !child->isAttached( ))
244  {
245  LBASSERT( !isMaster( ));
246  co::LocalNodePtr localNode = child->getConfig()->getLocalNode();
247  PKG packet( localNode->registerRequest( ));
248 
249  co::NodePtr node = child->getServer().get();
250  sender->send( node, packet );
251 
252  uint128_t identifier;
253  localNode->waitRequest( packet.requestID, identifier );
254  LBCHECK( localNode->mapObject( child,identifier,co::VERSION_NONE ));
255  }
256  child->commit( incarnation );
257  }

298  template< class P, class C >
299  inline void Object::releaseChildren( const std::vector< C* >& children )
300  {
301  for( size_t i = children.size(); i > 0; --i )
302  {
303  C* child = children[ i - 1 ];
304 
305  if( child->isAttached( ))
306  {
307  getLocalNode()->releaseObject( child );
308  if( !isMaster( ))
309  {
310  static_cast< P* >( this )->_removeChild( child );
311  static_cast< P* >( this )->release( child );
312  }
313  }
314  else
315  {
316  LBASSERT( isMaster( ));
317  }
318  }
319  }
