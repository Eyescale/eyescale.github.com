33 template< typename T >
34 typename boost::enable_if< boost::is_integral<T> >::type
35 DataOStreamArchive::save( const T& t )
36 {
37 #if BOOST_VERSION < 104800
38  namespace bs = boost::detail;
39 #else
40  namespace bs = boost::spirit::detail;
41 #endif
42 
43  if( T temp = t )
44  {
45  // examine the number of bytes
46  // needed to represent the number
47  signed char size = 0;
48  do
49  {
50  temp >>= CHAR_BIT;
51  ++size;
52  }
53  while( temp != 0 && temp != (T) -1 );
54 
55  // encode the sign bit into the size
56  _saveSignedChar( t > 0 ? size : -size );
57  BOOST_ASSERT( t > 0 || boost::is_signed<T>::value) ;
58 
59  // we choose to use little endian because this way we just
60  // save the first size bytes to the stream and skip the rest
61  bs::store_little_endian<T, sizeof(T)>( &temp, t );
62  save_binary( &temp, size );
63  }
64  else
65  // zero optimization
66  _saveSignedChar (0 );
67 }

69 template< typename T >
70 typename boost::enable_if< boost::is_floating_point<T> >::type
71 DataOStreamArchive::save( const T& t )
72 {
73  namespace fp = boost::spirit::math;
74 
75  typedef typename fp::detail::fp_traits<T>::type traits;
76 
77  // if the no_infnan flag is set we must throw here
78  if( get_flags() & no_infnan && !fp::isfinite( t ))
79  throw DataStreamArchiveException( t );
80 
81  // if you end here there are three possibilities:
82  // 1. you're serializing a long double which is not portable
83  // 2. you're serializing a double but have no 64 bit integer
84  // 3. your machine is using an unknown floating point format
85  // after reading the note above you still might decide to
86  // deactivate this static assert and try if it works out.
87  typename traits::bits bits;
88  BOOST_STATIC_ASSERT( sizeof(bits) == sizeof(T));
89  BOOST_STATIC_ASSERT( std::numeric_limits<T>::is_iec559 );
90 
91  // examine value closely
92  switch( fp::fpclassify( t ))
93  {
94  case FP_ZERO:
95  bits = 0;
96  break;
97  case FP_NAN:
98  bits = traits::exponent | traits::mantissa;
99  break;
100  case FP_INFINITE:
101  bits = traits::exponent | (t<0) * traits::sign;
102  break;
103  case FP_SUBNORMAL:
104  assert( std::numeric_limits<T>::has_denorm );
105  case FP_NORMAL:
106  traits::get_bits(t, bits);
107  break;
108  default:
109  throw DataStreamArchiveException( t );
110  }
111 
112  save( bits );
113 }
