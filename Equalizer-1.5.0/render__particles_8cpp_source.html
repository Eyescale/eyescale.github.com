82 void ParticleRenderer::_drawPoints(bool color)
83 {
84  if (!m_pbo)

87  {
88  int k = 0;

91  glVertex3fv(&m_pos[k]);

93  }
94  }
95  glEnd();

97  else
98  {
99  glEnableClientState(GL_VERTEX_ARRAY);
100 
101  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_pbo);
102  glVertexPointer(4, GL_FLOAT, 0, 0);
103 
104  if (color)
105  {
106  glEnableClientState(GL_COLOR_ARRAY);
107  glBindBufferARB(GL_ARRAY_BUFFER_ARB, m_vboColor);
108  glColorPointer(4, GL_FLOAT, 0, 0);
109  }
110 
111  glDrawArrays(GL_POINTS, 0, m_numParticles);
112  glBindBufferARB(GL_ARRAY_BUFFER_ARB, 0);
113  glDisableClientState(GL_VERTEX_ARRAY);
114 
115  if (color)

255  inline float evalHermite(float pA, float pB, float vA, float vB, float u)
256  {
257  float u2=(u*u), u3=u2*u;
258  float B0 = 2*u3 - 3*u2 + 1;
259  float B1 = -2*u3 + 3*u2;
260  float B2 = u3 - 2*u2 + u;
261  float B3 = u3 - u;
262  return( B0*pA + B1*pB + B2*vA + B3*vB );
263  }

265  unsigned char* createGaussianMap(int N)
266  {
267  float *M = new float[2*N*N];
268  unsigned char *B = new unsigned char[4*N*N];
269  float X,Y,Y2,Dist;
270  float Incr = 2.0f/N;
271  int i=0;
272  int j = 0;
273  Y = -1.0f;
274  //float mmax = 0;
275  for (int y=0; y<N; y++, Y+=Incr)
276  {
277  Y2=Y*Y;
278  X = -1.0f;
279  for (int x=0; x<N; x++, X+=Incr, i+=2, j+=4)
280  {
281  Dist = (float)sqrtf(X*X+Y2);
282  if (Dist>1) Dist=1;
283  M[i+1] = M[i] = evalHermite(1.0f,0,0,0,Dist);
284  B[j+3] = B[j+2] = B[j+1] = B[j] = (unsigned char)(M[i] * 255);
285  }
286  }
287  delete [] M;
288  return(B);
289  }
