125 {
126  if( stopRendering( ))
127  return;
128 
129  _initJitter();
130  if( _isDone( ))
131  return;
132 
133  Window* window = static_cast< Window* >( getWindow( ));
134  VertexBufferState& state = window->getState();
135  const Model* oldModel = _model;
136  const Model* model = _getModel();
137 
138  if( oldModel != model )
139  state.setFrustumCulling( false ); // create all display lists/VBOs
140 
141  if( model )

144  eq::Channel::frameDraw( frameID ); // Setup OpenGL state

188  Accum& accum = _accum[ lunchbox::getIndexOfLastBit( getEye()) ];
189  accum.stepsDone = LB_MAX( accum.stepsDone,
190  getSubPixel().size * getPeriod( ));
191  accum.transfer = true;

194 void Channel::frameAssemble( const eq::uint128_t& frameID )
195 {
196  if( stopRendering( ))
197  return;
198 
199  if( _isDone( ))
200  return;
201 
202  Accum& accum = _accum[ lunchbox::getIndexOfLastBit( getEye()) ];
203 
204  if( getPixelViewport() != _currentPVP )
205  {
206  accum.transfer = true;
207 
208  if( accum.buffer && !accum.buffer->usesFBO( ))
209  {
210  LBWARN << "Current viewport different from view viewport, "
211  << "idle anti-aliasing not implemented." << std::endl;
212  accum.step = 0;
213  }
214 
215  eq::Channel::frameAssemble( frameID );
216  return;
217  }
218  // else
219 
220  accum.transfer = true;
221  const eq::Frames& frames = getInputFrames();
222 
223  for( eq::Frames::const_iterator i = frames.begin(); i != frames.end(); ++i )
224  {
225  eq::Frame* frame = *i;
226  const eq::SubPixel& curSubPixel = frame->getSubPixel();
227 
228  if( curSubPixel != eq::SubPixel::ALL )
229  accum.transfer = false;
230 
231  accum.stepsDone = LB_MAX( accum.stepsDone, frame->getSubPixel().size *
232  frame->getPeriod( ));
233  }
234 
235  applyBuffer();
236  applyViewport();
237  setupAssemblyState();
238 
239  try
240  {
241  eq::Compositor::assembleFrames( getInputFrames(), this, accum.buffer );
242  }
243  catch( const co::Exception& e )
244  {
245  LBWARN << e.what() << std::endl;
246  }
247 
248  resetAssemblyState();
249 }

300 void Channel::frameFinish( const eq::uint128_t& frameID,
301  const uint32_t frameNumber )
302 {
303  if( stopRendering( ))
304  return;
305 
306  for( size_t i = 0; i < eq::NUM_EYES; ++i )
307  {
308  Accum& accum = _accum[ i ];
309  if( accum.step > 0 )
310  {

321 void Channel::frameViewFinish( const eq::uint128_t& frameID )
322 {
323  if( stopRendering( ))
324  return;
325 
326  applyBuffer();
327 
328  const FrameData& frameData = _getFrameData();
329  Accum& accum = _accum[ lunchbox::getIndexOfLastBit( getEye()) ];
330 
331  if( accum.buffer )
332  {
333  const eq::PixelViewport& pvp = getPixelViewport();
334  const bool isResized = accum.buffer->resize( pvp.w, pvp.h );
335 
336  if( isResized )
337  {
338  const View* view = static_cast< const View* >( getView( ));
339  accum.buffer->clear();
340  accum.step = view->getIdleSteps();
341  accum.stepsDone = 0;
342  }
343  else if( frameData.isIdle( ))
344  {
345  setupAssemblyState();
346 
347  if( !_isDone() && accum.transfer )
348  accum.buffer->accum();
349  accum.buffer->display();
350 
351  resetAssemblyState();
352  }
353  }
354 
355  applyViewport();
356  _drawOverlay();
357  _drawHelp();
358 
359  if( frameData.useStatistics())
360  drawStatistics();
361 
362  ConfigEvent event;
363  event.data.originator = getID();
364  event.data.type = ConfigEvent::IDLE_AA_LEFT;
365 
366  if( frameData.isIdle( ))
367  {
368  event.steps = 0;
369  for( size_t i = 0; i < eq::NUM_EYES; ++i )
370  event.steps = LB_MAX( event.steps, _accum[i].step );
371  }
372  else
373  {
374  const View* view = static_cast< const View* >( getView( ));
375  event.steps = view ? view->getIdleSteps() : 0;
376  }
377 
378  // if _jitterStep == 0 and no user redraw event happened, the app will exit
379  // FSAA idle mode and block on the next redraw event.
380  eq::Config* config = getConfig();
381  config->sendEvent( event );
382 }

407 void Channel::_initJitter()
408 {
409  if( !_initAccum( ))
410  return;
411 
412  const FrameData& frameData = _getFrameData();
413  if( frameData.isIdle( ))
414  return;
415 
416  const View* view = static_cast< const View* >( getView( ));
417  if( !view )
418  return;
419 
420  const int32_t idleSteps = view->getIdleSteps();
421  if( idleSteps == 0 )
422  return;
423 
424  // ready for the next FSAA
425  Accum& accum = _accum[ lunchbox::getIndexOfLastBit( getEye()) ];
426  if( accum.buffer )
427  accum.buffer->clear();
428  accum.step = idleSteps;
429 }

431 bool Channel::_initAccum()
432 {
433  View* view = static_cast< View* >( getNativeView( ));
434  if( !view ) // Only alloc accum for dest
435  return true;
436 
437  const eq::Eye eye = getEye();
438  Accum& accum = _accum[ lunchbox::getIndexOfLastBit( eye ) ];
439 
440  if( accum.buffer ) // already done
441  return true;
442 
443  if( accum.step == -1 ) // accum init failed last time
444  return false;
445 
446  // Check unsupported cases
447  if( !eq::util::Accum::usesFBO( glewGetContext( )))
448  {
449  for( size_t i = 0; i < eq::NUM_EYES; ++i )
450  {
451  if( _accum[ i ].buffer )
452  {
453  LBWARN << "glAccum-based accumulation does not support "
454  << "stereo, disabling idle anti-aliasing."
455  << std::endl;
456  for( size_t j = 0; j < eq::NUM_EYES; ++j )
457  {
458  delete _accum[ j ].buffer;
459  _accum[ j ].buffer = 0;
460  _accum[ j ].step = -1;
461  }
462 
463  view->setIdleSteps( 0 );
464  return false;
465  }
466  }
467  }
468 
469  // set up accumulation buffer
470  accum.buffer = new eq::util::Accum( glewGetContext( ));
471  const eq::PixelViewport& pvp = getPixelViewport();
472  LBASSERT( pvp.isValid( ));
473 
474  if( !accum.buffer->init( pvp, getWindow()->getColorFormat( )) ||
475  accum.buffer->getMaxSteps() < 256 )
476  {
477  LBWARN <<"Accumulation buffer initialization failed, "
478  << "idle AA not available." << std::endl;
479  delete accum.buffer;
480  accum.buffer = 0;
481  accum.step = -1;
482  return false;
483  }
484 
485  // else
486  LBVERB << "Initialized "
487  << (accum.buffer->usesFBO() ? "FBO accum" : "glAccum")
488  << " buffer for " << getName() << " " << getEye()
489  << std::endl;
490 
491  view->setIdleSteps( accum.buffer ? 256 : 0 );
492  return true;
493 }

500 eq::Vector2f Channel::getJitter() const
501 {
502  const FrameData& frameData = _getFrameData();
503  const Accum& accum = _accum[ lunchbox::getIndexOfLastBit( getEye()) ];
504 
505  if( !frameData.isIdle() || accum.step <= 0 )
506  return eq::Channel::getJitter();
507 
508  const View* view = static_cast< const View* >( getView( ));
509  if( !view || view->getIdleSteps() != 256 )
510  return eq::Vector2f::ZERO;
511 
512  const eq::Vector2i jitterStep = _getJitterStep();
513  if( jitterStep == eq::Vector2i::ZERO )
514  return eq::Vector2f::ZERO;
515 
516  const eq::PixelViewport& pvp = getPixelViewport();
517  const float pvp_w = float( pvp.w );
518  const float pvp_h = float( pvp.h );
519  const float frustum_w = float(( getFrustum().get_width( )));
520  const float frustum_h = float(( getFrustum().get_height( )));
521 
522  const float pixel_w = frustum_w / pvp_w;
523  const float pixel_h = frustum_h / pvp_h;
524 
525  const float sampleSize = 16.f; // sqrt( 256 )
526  const float subpixel_w = pixel_w / sampleSize;
527  const float subpixel_h = pixel_h / sampleSize;
528 
529  // Sample value randomly computed within the subpixel
530  lunchbox::RNG rng;
531  const eq::Pixel& pixel = getPixel();
532 
533  const float i = ( rng.get< float >() * subpixel_w +
534  float( jitterStep.x( )) * subpixel_w ) / float( pixel.w );
535  const float j = ( rng.get< float >() * subpixel_h +
536  float( jitterStep.y( )) * subpixel_h ) / float( pixel.h );
537 
538  return eq::Vector2f( i, j );
539 }

551 eq::Vector2i Channel::_getJitterStep() const
552 {
553  const eq::SubPixel& subPixel = getSubPixel();
554  const uint32_t channelID = subPixel.index;
555  const View* view = static_cast< const View* >( getView( ));
556  if( !view )
557  return eq::Vector2i::ZERO;
558 
559  const uint32_t totalSteps = uint32_t( view->getIdleSteps( ));
560  if( totalSteps != 256 )
561  return eq::Vector2i::ZERO;
562 
563  const Accum& accum = _accum[ lunchbox::getIndexOfLastBit( getEye()) ];
564  const uint32_t subset = totalSteps / getSubPixel().size;
565  const uint32_t index = ( accum.step * _primes[ channelID % 100 ] )%subset +
566  ( channelID * subset );
567  const uint32_t sampleSize = 16;
568  const int dx = index % sampleSize;
569  const int dy = index / sampleSize;
570 
571  return eq::Vector2i( dx, dy );
572 }

593 void Channel::_drawModel( const Model* scene )
594 {
595  Window* window = static_cast< Window* >( getWindow( ));
596  VertexBufferState& state = window->getState();
597  const FrameData& frameData = _getFrameData();
598 
599  if( frameData.getColorMode() == COLOR_MODEL && scene->hasColors( ))
600  state.setColors( true );
601  else
602  state.setColors( false );
603  state.setChannel( this );
604 
605  // Compute cull matrix
606  const eq::Matrix4f& rotation = frameData.getCameraRotation();
607  const eq::Matrix4f& modelRotation = frameData.getModelRotation();
608  eq::Matrix4f position = eq::Matrix4f::IDENTITY;
609  position.set_translation( frameData.getCameraPosition());
610 
611  const eq::Frustumf& frustum = getFrustum();
612  const eq::Matrix4f projection = useOrtho() ? frustum.compute_ortho_matrix():
613  frustum.compute_matrix();
614  const eq::Matrix4f& view = getHeadTransform();
615  const eq::Matrix4f model = rotation * position * modelRotation;
616 
617  state.setProjectionModelViewMatrix( projection * view * model );
618  state.setRange( &getRange().start);
619 
620  const eq::Pipe* pipe = getPipe();
621  const GLuint program = state.getProgram( pipe );
622  if( program != VertexBufferState::INVALID )
623  glUseProgram( program );
624 
625  scene->cullDraw( state );
626 
627  state.setChannel( 0 );
628  if( program != VertexBufferState::INVALID )
629  glUseProgram( 0 );
630 
631  const InitData& initData =
632  static_cast<Config*>( getConfig( ))->getInitData();
633  if( !initData.useROI( ))
634  {
635  declareRegion( getPixelViewport( ));
636  return;
637  }
638 
639 #ifndef NDEBUG // region border
640  const eq::PixelViewport& pvp = getPixelViewport();
641  const eq::PixelViewport& region = getRegion();
642 
643  glMatrixMode( GL_PROJECTION );
644  glLoadIdentity();
645  glOrtho( 0.f, pvp.w, 0.f, pvp.h, -1.f, 1.f );
646  glMatrixMode( GL_MODELVIEW );
647  glLoadIdentity();
648 
649  const eq::View* currentView = getView();
650  if( frameData.getColorMode() == COLOR_DEMO )
651  {
652  const eq::Vector3ub color = getUniqueColor();
653  glColor3ub( color.r(), color.g(), color.b() );
654  }
655  else if( currentView &&
656  frameData.getCurrentViewID() == currentView->getID( ))
657  {
658  glColor3f( 0.f, 0.f, 0.f );
659  }
660  else
661  glColor3f( 1.f, 1.f, 1.f );

806 void Channel::_updateNearFar( const mesh::BoundingSphere& boundingSphere )
807 {
808  // compute dynamic near/far plane of whole model
809  const FrameData& frameData = _getFrameData();
810 
811  const eq::Matrix4f& rotation = frameData.getCameraRotation();
812  const eq::Matrix4f headTransform = getHeadTransform() * rotation;
813 
814  eq::Matrix4f modelInv;
815  compute_inverse( headTransform, modelInv );
816 
817  const eq::Vector3f zero = modelInv * eq::Vector3f::ZERO;
818  eq::Vector3f front = modelInv * eq::Vector3f( 0.0f, 0.0f, -1.0f );
819 
820  front -= zero;
821  front.normalize();
822  front *= boundingSphere.w();
823 
824  const eq::Vector3f center =
825  frameData.getCameraPosition().get_sub_vector< 3 >() -
826  boundingSphere.get_sub_vector< 3 >();
827  const eq::Vector3f nearPoint = headTransform * ( center - front );
828  const eq::Vector3f farPoint = headTransform * ( center + front );
829 
830  if( useOrtho( ))
831  {
832  LBASSERTINFO( fabs( farPoint.z() - nearPoint.z() ) >
833  std::numeric_limits< float >::epsilon(),
834  nearPoint << " == " << farPoint );
835  setNearFar( -nearPoint.z(), -farPoint.z() );
836  }
837  else
838  {
839  // estimate minimal value of near plane based on frustum size
840  const eq::Frustumf& frustum = getFrustum();
841  const float width = fabs( frustum.right() - frustum.left() );
842  const float height = fabs( frustum.top() - frustum.bottom() );
843  const float size = LB_MIN( width, height );
844  const float minNear = frustum.near_plane() / size * .001f;
845 
846  const float zNear = LB_MAX( minNear, -nearPoint.z() );
847  const float zFar = LB_MAX( zNear * 2.f, -farPoint.z() );
848 
849  setNearFar( zNear, zFar );
850  }
851 }
