45 int OSGScaleViewer::run()
46 {
47  // 1. connect to server
48  eq::ServerPtr server = new eq::Server();
49  if( !connectServer( server ))
50  {
51  LBERROR << "Can't open server" << std::endl;
52  return EXIT_FAILURE;
53  }
54 
55  // 2. choose config
56  eq::ConfigParams configParams;
57  Config* config = static_cast<Config*>( server->chooseConfig( configParams));
58 
59  if( !config )
60  {
61  LBERROR << "No matching config on server" << std::endl;
62  disconnectServer( server );
63  return EXIT_FAILURE;
64  }
65 
66  config->setInitData( _initData );
67  if( !config->init( ))
68  {
69  server->releaseConfig( config );
70  disconnectServer( server );
71  return EXIT_FAILURE;
72  }
73  else if( config->getError( ))
74  LBWARN << "Error during initialization: " << config->getError()
75  << std::endl;
76 
77  // 4. run main loop
78  while( config->isRunning( ))
79  {
80  config->startFrame();
81  config->finishFrame();
82  }
83  config->finishAllFrames();
84 
85  // 5. exit config
86  config->exit();
87 
88  // 6. cleanup and exit
89  server->releaseConfig( config );
90  if( !disconnectServer( server ))
91  LBERROR << "Client::disconnectServer failed" << std::endl;
92 
93  server = 0;
94 
95  return EXIT_SUCCESS;
96 }
