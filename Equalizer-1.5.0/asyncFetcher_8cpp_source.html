40 static eq::SystemWindow* initSharedContextWindow( eq::Window* window )
41 {
42  LBASSERT( window );
43 
44  // store old drawable of window and set window's drawable to OFF,
45  // create another (shared) osWindow and restore original drawable
46  const int32_t drawable =
47  window->getIAttribute( eq::Window::IATTR_HINT_DRAWABLE );
48  window->setIAttribute( eq::Window::IATTR_HINT_DRAWABLE, eq::OFF );
49 
50  const eq::Pipe* pipe = window->getPipe();
51  LBASSERT( pipe );
52 
53  eq::SystemWindow* sharedWindow =
54  pipe->getWindowSystem().createWindow( window );
55 
56  if( sharedWindow )
57  {
58  if( sharedWindow->configInit( ))
59  sharedWindow->makeCurrent();
60  else
61  {
62  LBWARN << "OS Window initialization failed: " << std::endl;
63  delete sharedWindow;
64  sharedWindow = 0;
65  }
66  }
67  else
68  {
69  LBERROR << "Failed to create shared context window for "
70  << pipe->getWindowSystem() << std::endl;
71  }
72 
73  window->setIAttribute( eq::Window::IATTR_HINT_DRAWABLE, drawable );
74 
75  LBINFO << "Async fetcher initialization finished" << std::endl;
76  return sharedWindow;
77 }

80 static void deleteSharedContextWindow( eq::Window* window,
81  eq::SystemWindow** sharedWindow,
82  eq::ObjectManager** objectManager )
83 {
84  LBWARN << "Deleting shared context" << std::endl;
85  if( !sharedWindow || !*sharedWindow )
86  return;
87 
88  if( *objectManager )
89  {
90  (*objectManager)->deleteAll();
91  delete *objectManager;
92  *objectManager = 0;
93  }
94 
95  (*sharedWindow)->configExit(); // mb set window to 0 before that?
96 
97  delete *sharedWindow;
98  *sharedWindow = 0;
99 }

131 void AsyncFetcher::run()
132 {
133  LBASSERT( !_sharedWindow );
134  _sharedWindow = initSharedContextWindow( _window );
135  _outQueue.push( TextureId( )); // unlock pipe thread
136  if( !_sharedWindow )
137  return;
138 
139  _objectManager = new eq::ObjectManager( glewGetContext( ));
140  LBINFO << "async fetcher initialized: " << _window << std::endl;
141 
142  uint8_t* i = 0;
143  bool running = true;
144  lunchbox::sleep( 1000 ); // imitate loading of the first texture
145  while( running )
146  {
147  // generate new texture
148  eq::util::Texture* tx = _objectManager->newEqTexture( ++i,
149  GL_TEXTURE_2D );
150  tx->init( GL_RGBA8, 64, 64 );
151 
152  int j = 0;
153  lunchbox::RNG rng;
154  for( int y = 0; y < 64; ++y )
155  {
156  for( int x = 0; x < 64; ++x )
157  {
158  const GLbyte rnd = rng.get< uint8_t >() % 127;
159  const GLbyte val = (x / 8) % 2 == (y / 8) % 2 ? rnd : 0;
160  _tmpTexture[ j++ ] = val;
161  _tmpTexture[ j++ ] = val;
162  _tmpTexture[ j++ ] = val;
163  _tmpTexture[ j++ ] = val;
164  }
165  }
166  tx->upload( 64, 64, _tmpTexture );
167  EQ_GL_CALL( glFinish( ));
168 
169  // add new texture to the pool
170  _outQueue.push( TextureId( tx->getName(), i ));
171 
172  // imitate hard work of loading something else
173  lunchbox::sleep( rng.get< uint32_t >() % 5000u );
174 
175  // clean unused textures
176  const void* keyToDelete = 0;
177  while( _inQueue.tryPop( keyToDelete ))
178  {
179  if( keyToDelete )
180  {
181  LBWARN << "Deleting eq texture " << keyToDelete << std::endl;
182  _objectManager->deleteEqTexture( keyToDelete );
183  }
184  else
185  running = false;
186  }
187  }
188  deleteSharedContextWindow( _window, &_sharedWindow, &_objectManager );
189 }
