98 void SceneView::setDefaults(unsigned int options)
99 {
100  osg::CullSettings::setDefaults();
101 
102  _camera->getProjectionMatrix().makePerspective(50.0f,1.4f,1.0f,10000.0f);
103  _camera->getViewMatrix().makeIdentity();
104 
105  if (!_globalStateSet) _globalStateSet = new osg::StateSet;
106  else _globalStateSet->clear();
107 
108  if ((options & HEADLIGHT) || (options & SKY_LIGHT))
109  {
110  #if defined(OSG_GL_FIXED_FUNCTION_AVAILABLE)
111  _lightingMode=(options&HEADLIGHT) ? HEADLIGHT : SKY_LIGHT;
112  _light = new osg::Light;
113  _light->setLightNum(0);
114  _light->setAmbient(Vec4(0.00f,0.0f,0.00f,1.0f));
115  _light->setDiffuse(Vec4(0.8f,0.8f,0.8f,1.0f));
116  _light->setSpecular(Vec4(1.0f,1.0f,1.0f,1.0f));
117 
118 
119  _globalStateSet->setAssociatedModes(_light.get(),osg::StateAttribute::ON);
120 
121  // enable lighting by default.
122  _globalStateSet->setMode(GL_LIGHTING, osg::StateAttribute::ON);
123  #endif
124 
125  #if !defined(OSG_GLES1_AVAILABLE) && !defined(OSG_GLES2_AVAILABLE) && defined(OSG_GL_FIXED_FUNCTION_AVAILABLE)
126  osg::LightModel* lightmodel = new osg::LightModel;
127  lightmodel->setAmbientIntensity(osg::Vec4(0.1f,0.1f,0.1f,1.0f));
128  _globalStateSet->setAttributeAndModes(lightmodel, osg::StateAttribute::ON);
129  #endif
130  }
131  else
132  {
133  _lightingMode = NO_SCENEVIEW_LIGHT;
134  }
135 
136  _renderInfo.setState(new State);
137 
138  _stateGraph = new StateGraph;
139  _renderStage = new RenderStage;
140 
141 
142  if (options & COMPILE_GLOBJECTS_AT_INIT)
143  {
144  GLObjectsVisitor::Mode dlvMode = GLObjectsVisitor::COMPILE_DISPLAY_LISTS |
145  GLObjectsVisitor::COMPILE_STATE_ATTRIBUTES |
146  GLObjectsVisitor::CHECK_BLACK_LISTED_MODES;
147 
148  #ifdef __sgi
149  dlvMode = GLObjectsVisitor::COMPILE_STATE_ATTRIBUTES;
150  #endif
151 
152  // sgi's IR graphics has a problem with lighting and display lists, as it seems to store
153  // lighting state with the display list, and the display list visitor doesn't currently apply
154  // state before creating display lists. So will disable the init visitor default, this won't
155  // affect functionality since the display lists will be created as and when needed.
156  GLObjectsVisitor* dlv = new GLObjectsVisitor(dlvMode);
157  dlv->setNodeMaskOverride(0xffffffff);
158  _initVisitor = dlv;
159 
160  }
161 
162  _updateVisitor = new UpdateVisitor;
163 
164  _cullVisitor = CullVisitor::create();
165 
166  _cullVisitor->setStateGraph(_stateGraph.get());
167  _cullVisitor->setRenderStage(_renderStage.get());
168 
169  _globalStateSet->setGlobalDefaults();
170 
171  #if defined(OSG_GL_FIXED_FUNCTION_AVAILABLE)
172  // set up an texture environment by default to speed up blending operations.
173  osg::TexEnv* texenv = new osg::TexEnv;
174  texenv->setMode(osg::TexEnv::MODULATE);
175  _globalStateSet->setTextureAttributeAndModes(0,texenv, osg::StateAttribute::ON);
176  #endif
177 
178  _camera->setClearColor(osg::Vec4(0.2f, 0.2f, 0.4f, 1.0f));
179 }

241 void SceneView::updateUniforms()
242 {
243  if (!_localStateSet)
244  {
245  _localStateSet = new osg::StateSet;
246  }
247 
248  if (!_localStateSet) return;
249 
250  if ((_activeUniforms & FRAME_NUMBER_UNIFORM) && _frameStamp.valid())
251  {
252 #if OSG_MIN_VERSION_REQUIRED(2,9,11)
253  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_FrameNumber",osg::Uniform::UNSIGNED_INT);
254 #else
255  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_FrameNumber",osg::Uniform::INT);
256 #endif
257 
258  uniform->set(_frameStamp->getFrameNumber());
259  }
260 
261  if ((_activeUniforms & FRAME_TIME_UNIFORM) && _frameStamp.valid())
262  {
263  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_FrameTime",osg::Uniform::FLOAT);
264  uniform->set(static_cast<float>(_frameStamp->getReferenceTime()));
265  }
266 
267  if ((_activeUniforms & DELTA_FRAME_TIME_UNIFORM) && _frameStamp.valid())
268  {
269  float delta_frame_time = (_previousFrameTime != 0.0) ? static_cast<float>(_frameStamp->getReferenceTime()-_previousFrameTime) : 0.0f;
270  _previousFrameTime = _frameStamp->getReferenceTime();
271 
272  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_DeltaFrameTime",osg::Uniform::FLOAT);
273  uniform->set(delta_frame_time);
274  }
275 
276  if ((_activeUniforms & SIMULATION_TIME_UNIFORM) && _frameStamp.valid())
277  {
278  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_SimulationTime",osg::Uniform::FLOAT);
279  uniform->set(static_cast<float>(_frameStamp->getSimulationTime()));
280  }
281 
282  if ((_activeUniforms & DELTA_SIMULATION_TIME_UNIFORM) && _frameStamp.valid())
283  {
284  float delta_simulation_time = (_previousSimulationTime != 0.0) ? static_cast<float>(_frameStamp->getSimulationTime()-_previousSimulationTime) : 0.0f;
285  _previousSimulationTime = _frameStamp->getSimulationTime();
286 
287  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_DeltaSimulationTime",osg::Uniform::FLOAT);
288  uniform->set(delta_simulation_time);
289  }
290 
291  if (_activeUniforms & VIEW_MATRIX_UNIFORM)
292  {
293  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_ViewMatrix",osg::Uniform::FLOAT_MAT4);
294  uniform->set(getViewMatrix());
295  }
296 
297  if (_activeUniforms & VIEW_MATRIX_INVERSE_UNIFORM)
298  {
299  osg::Uniform* uniform = _localStateSet->getOrCreateUniform("osg_ViewMatrixInverse",osg::Uniform::FLOAT_MAT4);
300  uniform->set(osg::Matrix::inverse(getViewMatrix()));
301  }

336 void SceneView::inheritCullSettings(const osg::CullSettings& settings, unsigned int inheritanceMask)
337 {
338  if (_camera.valid() && _camera->getView())
339  {
340  if (inheritanceMask & osg::CullSettings::LIGHTING_MODE)
341  {
342  LightingMode newLightingMode = _lightingMode;
343 
344  switch(_camera->getView()->getLightingMode())
345  {
346  case(osg::View::NO_LIGHT): newLightingMode = NO_SCENEVIEW_LIGHT; break;
347  case(osg::View::HEADLIGHT): newLightingMode = HEADLIGHT; break;
348  case(osg::View::SKY_LIGHT): newLightingMode = SKY_LIGHT; break;
349  }
350 
351  if (newLightingMode != _lightingMode)
352  {
353  setLightingMode(newLightingMode);
354  }
355  }
356 
357  if (inheritanceMask & osg::CullSettings::LIGHT)
358  {
359  setLight(_camera->getView()->getLight());
360  }
361  }
362 
363  osg::CullSettings::inheritCullSettings(settings, inheritanceMask);
364 }

367 void SceneView::cull()
368 {
369  _dynamicObjectCount = 0;
370 
371  if (_camera->getNodeMask()==0) return;
372 
373  _renderInfo.setView(_camera->getView());
374 
375  // update the active uniforms
376  updateUniforms();
377 
378  if (!_renderInfo.getState())
379  {
380  osg::notify(osg::INFO) << "Warning: no valid osgUtil::SceneView::_state attached, creating a default state automatically."<< std::endl;
381 
382  // note the constructor for osg::State will set ContextID to 0 which will be fine to single context graphics
383  // applications which is ok for most apps, but not multiple context/pipe applications.
384  _renderInfo.setState(new osg::State);
385  }
386 
387  osg::State* state = _renderInfo.getState();
388 
389  if (!_localStateSet)
390  {
391  _localStateSet = new osg::StateSet;
392  }
393 
394  // we in theory should be able to be able to bypass reset, but we'll call it just incase.
395  //_state->reset();
396 
397  state->setFrameStamp(_frameStamp.get());
398  state->setDisplaySettings(_displaySettings.get());
399 
400 
401  if (!_cullVisitor)
402  {
403  osg::notify(osg::INFO) << "Warning: no valid osgUtil::SceneView:: attached, creating a default CullVisitor automatically."<< std::endl;
404  _cullVisitor = CullVisitor::create();
405  }
406  if (!_stateGraph)
407  {
408  osg::notify(osg::INFO) << "Warning: no valid osgUtil::SceneView:: attached, creating a global default StateGraph automatically."<< std::endl;
409  _stateGraph = new StateGraph;
410  }
411  if (!_renderStage)
412  {
413  osg::notify(osg::INFO) << "Warning: no valid osgUtil::SceneView::_renderStage attached, creating a default RenderStage automatically."<< std::endl;
414  _renderStage = new RenderStage;
415  }
416 
417  _cullVisitor->setTraversalMask(_cullMask);
418  bool computeNearFar = cullStage(getProjectionMatrix(),getViewMatrix(),_cullVisitor.get(),_stateGraph.get(),_renderStage.get(),getViewport());
419 
420  if (computeNearFar)
421  {
422  CullVisitor::value_type zNear = _cullVisitor->getCalculatedNearPlane();
423  CullVisitor::value_type zFar = _cullVisitor->getCalculatedFarPlane();
424  _cullVisitor->clampProjectionMatrix(getProjectionMatrix(),zNear,zFar);
425  }
426 }

428 bool SceneView::cullStage(const osg::Matrixd& projection,const osg::Matrixd& modelview,osgUtil::CullVisitor* cullVisitor, osgUtil::StateGraph* rendergraph, osgUtil::RenderStage* renderStage, osg::Viewport *viewport)
429 {
430 
431  if (!_camera || !viewport) return false;
432 
433  osg::ref_ptr<RefMatrix> proj = new osg::RefMatrix(projection);
434  osg::ref_ptr<RefMatrix> mv = new osg::RefMatrix(modelview);
435 
436  // collect any occluder in the view frustum.
437  if (_camera->containsOccluderNodes())
438  {
439  //std::cout << "Scene graph contains occluder nodes, searching for them"<<std::endl;
440 
441 
442  if (!_collectOccludersVisitor) _collectOccludersVisitor = new osg::CollectOccludersVisitor;
443 
444  _collectOccludersVisitor->inheritCullSettings(*this);
445 
446  _collectOccludersVisitor->reset();
447 
448  _collectOccludersVisitor->setFrameStamp(_frameStamp.get());
449 
450  // use the frame number for the traversal number.
451  if (_frameStamp.valid())
452  {
453  _collectOccludersVisitor->setTraversalNumber(_frameStamp->getFrameNumber());
454  }
455 
456  _collectOccludersVisitor->pushViewport(viewport);
457  _collectOccludersVisitor->pushProjectionMatrix(proj.get());
458  _collectOccludersVisitor->pushModelViewMatrix(mv.get(),osg::Transform::ABSOLUTE_RF);
459 
460  // traverse the scene graph to search for occluder in there new positions.
461  _collectOccludersVisitor->traverse(*_camera);
462 
463  _collectOccludersVisitor->popModelViewMatrix();
464  _collectOccludersVisitor->popProjectionMatrix();
465  _collectOccludersVisitor->popViewport();
466 
467  // sort the occluder from largest occluder volume to smallest.
468  _collectOccludersVisitor->removeOccludedOccluders();
469 
470 
471  osg::notify(osg::DEBUG_INFO) << "finished searching for occluder - found "<<_collectOccludersVisitor->getCollectedOccluderSet().size()<<std::endl;
472 
473  cullVisitor->getOccluderList().clear();
474  std::copy(_collectOccludersVisitor->getCollectedOccluderSet().begin(),_collectOccludersVisitor->getCollectedOccluderSet().end(), std::back_insert_iterator<CullStack::OccluderList>(cullVisitor->getOccluderList()));
475  }
476 
477 
478 
479  cullVisitor->reset();
480 
481  cullVisitor->setFrameStamp(_frameStamp.get());
482 
483  // use the frame number for the traversal number.
484  if (_frameStamp.valid())
485  {
486  cullVisitor->setTraversalNumber(_frameStamp->getFrameNumber());
487  }
488 
489  cullVisitor->inheritCullSettings(*this);
490 
491  cullVisitor->setStateGraph(rendergraph);
492  cullVisitor->setRenderStage(renderStage);
493 
494  cullVisitor->setRenderInfo( _renderInfo );
495 
496  renderStage->reset();
497 
498  // comment out reset of rendergraph since clean is more efficient.
499  // rendergraph->reset();
500 
501  // use clean of the rendergraph rather than reset, as it is able to
502  // reuse the structure on the rendergraph in the next frame. This
503  // achieves a certain amount of frame cohereancy of memory allocation.
504  rendergraph->clean();
505 
506  renderStage->setViewport(viewport);
507  renderStage->setClearColor(_camera->getClearColor());
508  renderStage->setClearDepth(_camera->getClearDepth());
509  renderStage->setClearAccum(_camera->getClearAccum());
510  renderStage->setClearStencil(_camera->getClearStencil());
511  renderStage->setClearMask(_camera->getClearMask());
512 
513 #if 1
514  renderStage->setCamera(_camera.get());
515 #endif
516 
517  #if defined(OSG_GL_FIXED_FUNCTION_AVAILABLE)
518  switch(_lightingMode)
519  {
520  case(HEADLIGHT):
521  if (_light.valid()) renderStage->addPositionedAttribute(NULL,_light.get());
522  else osg::notify(osg::WARN)<<"Warning: no osg::Light attached to ogUtil::SceneView to provide head light.*/"<<std::endl;
523  break;
524  case(SKY_LIGHT):
525  if (_light.valid()) renderStage->addPositionedAttribute(mv.get(),_light.get());
526  else osg::notify(osg::WARN)<<"Warning: no osg::Light attached to ogUtil::SceneView to provide sky light.*/"<<std::endl;
527  break;
528  default:
529  break;
530  }
531  #endif
532 
533  if (_globalStateSet.valid()) cullVisitor->pushStateSet(_globalStateSet.get());
534  if (_secondaryStateSet.valid()) cullVisitor->pushStateSet(_secondaryStateSet.get());
535  if (_localStateSet.valid()) cullVisitor->pushStateSet(_localStateSet.get());
536 
537 
538  cullVisitor->pushViewport(viewport);
539  cullVisitor->pushProjectionMatrix(proj.get());
540  cullVisitor->pushModelViewMatrix(mv.get(),osg::Transform::ABSOLUTE_RF);
541 
542  // traverse the scene graph to generate the rendergraph.
543  // If the camera has a cullCallback execute the callback which has the
544  // requirement that it must traverse the camera's children.
545  {
546  osg::NodeCallback* callback = _camera->getCullCallback();
547  if (callback) (*callback)(_camera.get(), cullVisitor);
548  else cullVisitor->traverse(*_camera);
549  }
550 
551 
552  cullVisitor->popModelViewMatrix();
553  cullVisitor->popProjectionMatrix();
554  cullVisitor->popViewport();
555 
556  if (_localStateSet.valid()) cullVisitor->popStateSet();
557  if (_secondaryStateSet.valid()) cullVisitor->popStateSet();
558  if (_globalStateSet.valid()) cullVisitor->popStateSet();
559 
560 
561  renderStage->sort();
562 
563  // prune out any empty StateGraph children.
564  // note, this would be not required if the rendergraph had been
565  // reset at the start of each frame (see top of this method) but
566  // a clean has been used instead to try to minimize the amount of
567  // allocation and deleteing of the StateGraph nodes.
568  rendergraph->prune();
569 
570  // set the number of dynamic objects in the scene.
571  _dynamicObjectCount += renderStage->computeNumberOfDynamicRenderLeaves();
572 
573 
574  bool computeNearFar = (cullVisitor->getComputeNearFarMode()!=osgUtil::CullVisitor::DO_NOT_COMPUTE_NEAR_FAR) && getSceneData()!=0;
575  return computeNearFar;
576 }

607 void SceneView::draw()
608 {
609  if (_camera->getNodeMask()==0) return;
610 
611  osg::State* state = _renderInfo.getState();
612  state->initializeExtensionProcs();
613 
614  osg::Texture::TextureObjectManager* tom = osg::Texture::getTextureObjectManager(state->getContextID()).get();
615  tom->newFrame(state->getFrameStamp());
616 
617  osg::GLBufferObjectManager* bom = osg::GLBufferObjectManager::getGLBufferObjectManager(state->getContextID()).get();
618  bom->newFrame(state->getFrameStamp());
619 
620  if (!_initCalled) init();
621 
622  // note, to support multi-pipe systems the deletion of OpenGL display list
623  // and texture objects is deferred until the OpenGL context is the correct
624  // context for when the object were originally created. Here we know what
625  // context we are in so can flush the appropriate caches.
626 
627  if (_requiresFlush)
628  {
629  double availableTime = 0.005;
630  flushDeletedGLObjects(availableTime);
631  }
632 
633  // assume the the draw which is about to happen could generate GL objects that need flushing in the next frame.
634  _requiresFlush = true;
635 
636  state->setInitialViewMatrix(new osg::RefMatrix(getViewMatrix()));
637 
638  RenderLeaf* previous = NULL;
639 
640  _localStateSet->setAttribute(getViewport());
641 
642  // bog standard draw.
643  _renderStage->drawPreRenderStages(_renderInfo,previous);
644  _renderStage->draw(_renderInfo,previous);
645 
646  // re apply the defalt OGL state.
647  state->popAllStateSets();
648  state->apply();
649 
650 #if 0
651  if (_camera->getPostDrawCallback())
652  {
653  (*(_camera->getPostDrawCallback()))(*_camera);
654  }
655 #endif
656 
657  if (state->getCheckForGLErrors()!=osg::State::NEVER_CHECK_GL_ERRORS)
658  {
